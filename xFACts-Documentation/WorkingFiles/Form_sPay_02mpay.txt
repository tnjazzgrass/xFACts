Private Sub procIntWO_Click()
Dim dB As Database
Dim rst1 As Recordset, rst2 As Recordset, rst4 As Recordset
Dim sql1 As String, sql2 As String, sql3 As String, sql4 As String
Dim recCnt As Integer, cpay As Currency, i As Currency, p As Currency

    Set dB = CurrentDb
    
    dB.Execute ("UPDATE wo_base_account SET NewPayPri = AccntDuePri;")
    
    '   Select all the CPJs to be evaluated
        sql1 = "SELECT * FROM wo_base_consumer WHERE xType = 'RPY' ORDER BY cnsmr_id;"
        
        Set rst1 = dB.OpenRecordset(sql1)
        
            Do Until rst1.EOF
            
            sql2 = "SELECT * FROM wo_base_account WHERE OrigCPJ_id = '" & rst1!OrigCPJ_id & "' AND xType = 'RPY' ORDER BY cnsmr_accnt_plcmnt, cnsmr_accnt_id;"
            Set rst2 = dB.OpenRecordset(sql2)
            
            sql3 = "SELECT DISTINCT OrigCPJ_id, SUM([NewPayPri]) AS rPd FROM wo_base_account WHERE OrigCPJ_id = '" & rst1!OrigCPJ_id & "' GROUP BY OrigCPJ_id;"
            Set rst3 = dB.OpenRecordset(sql3)
              
            cpay = rst1!OrigPayAmt - rst3!rPd
            
                Do Until rst2.EOF
    
                    'i = IIf(Round((rst2!AccntDueInt * 100), 2) / 100 <= cpay, Round((rst2!AccntDueInt * 100), 2) / 100, cpay)
                    i = IIf(Int(100 * rst2!AccntDueInt) / 100 <= cpay, Int(100 * rst2!AccntDueInt) / 100, cpay)
                    sql3 = "UPDATE wo_base_account SET NewPayInt = " & i & " WHERE cnsmr_accnt_id = " & rst2!cnsmr_accnt_id & ";"
                    dB.Execute (sql3)
                
                cpay = cpay - i
                rst2.MoveNext
                Loop
    
                i = 0
                cpay = 0
                
            rst1.MoveNext
            Loop
            
            dB.Execute ("UPDATE wo_base_account SET NewPayAmt = NewPayPri + NewPayInt;")
            
        rst1.MoveFirst
        
        Do Until rst1.EOF
        
            sql4 = "SELECT DISTINCT OrigCPJ_id, SUM([NewPayAmt]) AS rPd FROM wo_base_account WHERE OrigCPJ_id = '" & rst1!OrigCPJ_id & "' GROUP BY OrigCPJ_id;"
            Set rst4 = dB.OpenRecordset(sql4)
        
            dB.Execute ("UPDATE wo_base_consumer SET NewPayAmt = " & rst4!rPd & " WHERE OrigCPJ_id = '" & rst1!OrigCPJ_id & "' AND xType = 'RPY';")
            
        rst1.MoveNext
        Loop
    
        rst1.Close

        Select Case Me.runEnv
        
            Case 1
                
                intWO ("test")
            
            Case 2
            
                intWO ("stage")
            
            Case 3
                
                intWO ("prod")
            
        End Select
    
    dB.Close
    
End Sub
