Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.pay_bulk_Padj_02_dtl.Requery
    
End Sub

Private Sub postPlaceAdj_Click()
Dim dB As Database
Dim tgs As Recordset

    Set dB = CurrentDb
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM payment_info WHERE xProc = 'N' AND xType = 'PAD' AND xUser = '" & Me.usr.Value & "' AND PayConsumerNumber IS NOT NULL AND PayAccountNumber IS NOT NULL ;")
    
        If tgs!recs = 0 Then
        
            MsgBox "Please enter all required fields for adjustments before attempting to run the process.", vbCritical, "INCOMPLETE/MISSING DATA"
            
        Else

            If Me.JIRA = "SD-" Or Mid(Me.JIRA, 3, 1) <> "-" Then
                
                    MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                    
            Else
            
                dB.Execute ("UPDATE payment_info SET PayAmt = PayBucketAmt + PayBucket2Amt WHERE xProc = 'N' AND xType = 'PAD' AND xUser = '" & Me.usr.Value & "';")
                
                Select Case Me.runEnv
            
                    Case 1
                    
                        PlacementAdj ("test")
                
                    Case 2
                    
                        PlacementAdj ("stage")
                
                    Case 3
                    
                        PlacementAdj ("prod")
                        
                End Select
            
            End If
            
        End If
       
    Me.pay_bulk_Padj_02_dtl.Requery
    Me.JIRA.Value = "SD-"
    dB.Close

End Sub
