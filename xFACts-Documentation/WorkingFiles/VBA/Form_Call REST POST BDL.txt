Private Sub APImoveAcct_Click()
Dim del As Long
Dim dB As Database
Dim rst1 As Recordset
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String

    Set dB = CurrentDb
    Set rst1 = dB.OpenRecordset("SELECT * FROM rData WHERE Processed = 'N'")
    
    Do Until rst1.EOF
        
        jBody = "{"
        jBody = jBody & Chr(13) & Chr(10) & """consumerAgencyId"":" & rst1!ConsumerID & ","
        jBody = jBody & Chr(13) & Chr(10) & """accountIds"":[" & rst1!AccountID & "],"
        jBody = jBody & Chr(13) & Chr(10) & """attachConsumers"":false,"
        jBody = jBody & Chr(13) & Chr(10) & """attachLinkedConsumers"":false"
        jBody = jBody & Chr(13) & Chr(10) & "}"
    
            httpReq.Open "POST", Me.APIurl & "consumers/" & rst1!ConsumerID & "/split", False, Me.APIuser, Me.APIpass
            httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
            httpReq.setRequestHeader "Authorization", Me.authCode
            httpReq.sEnd (jBody)
        
       response = httpReq.responseText
       Debug.Print response
        
       sql = "UPDATE rData SET Processed = 'Y' WHERE ConsumerID = '" & rst1!ConsumerID & "' AND AccountID = '" & rst1!AccountID & "'"
       dB.Execute (sql)
       
'       Debug.Print Chr(13) & Chr(10) & sql
       
       del = Timer + 2
       Do While Timer < del
           DoEvents
       Loop
    
    rst1.MoveNext
    Loop
    
    MsgBox "The split consumer/account APIs have been submitted", vbInformation, "WEBSERVICE RESPONSE"

End Sub
