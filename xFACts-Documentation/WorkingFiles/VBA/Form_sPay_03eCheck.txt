Private Sub bckChkEdit_Click()

    DoCmd.Close acForm, Me.Form.Name, acSaveNo
    DoCmd.OpenForm "sPay_03_check", acNormal

End Sub
Private Sub delChkEdit_Click()
Dim dB As Database

    Set dB = CurrentDb
    dB.Execute ("Delete payment_info.PayID FROM payment_info WHERE (((payment_info.PayID)=" & Me.ePayID & "));")
    dB.Close
    
    MsgBox "This record has been deleted from the current batch.", vbOKOnly, "RECORD DELETED"
    
    DoCmd.Close acForm, Me.Form.Name, acSaveNo
    DoCmd.OpenForm "sPay_03_check", acNormal

End Sub

Private Sub updChkEdit_Click()
Dim sql As String
Dim err As Integer
Dim rec As Recordset
Dim dB As Database
Dim look As Recordset
Dim curr_bal As Double

'   Error Check - make sure form is filled out
    err = 0
    
    If IsNull(Me.eConsumer) = True Then
        MsgBox "Please enter a consumer number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.eCheckNo) = True Then
        MsgBox "Please enter a check number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.eAccount) = True Then
        MsgBox "Please enter an account number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.eAmount) = True Then
        MsgBox "Please enter a valid payment amount to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.eComment) = True Then
        MsgBox "Please enter a ticket number or payment comment to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If

    If err = 0 Then
    
        Set dB = CurrentDb
        
        Set look = dB.OpenRecordset("SELECT cnsmr_idntfr_agncy_id FROM dbo_vw_Apps Toolkit_cnsmr_accnts WHERE cnsmr_accnt_idntfr_agncy_id = '" & Me.eAccount & "';")
        
        If look.EOF Then
        
            MsgBox "This app cannot be used to process payments for this account. Please use Debt Manager to post payments on consumers with fewer than 100 accounts.", vbOKOnly, "NO RECORD FOUND"
            
        Else
            
            Set rec = dB.OpenRecordset("SELECT ACCOUNT_AGENCY_ID, account_curr_bal, account_status_tag FROM dbo_vw_Apps Toolkit_accnt_smmry WHERE ACCOUNT_AGENCY_ID = '" & Me.eAccount & "';")
            
                Select Case rec!account_status_tag
                    Case "TA_CSCLS", "TA_CSPIF", "TA_CSRET", "TA_CSRTR", "TA_CSSIF"
                        err = err + 1
                        MsgBox "This account has a status tag which makes it ineligible for payments. Please choose another account.", vbOKOnly, "ACCOUNT CLOSED"
                End Select
                
                    If err = 0 Then
                        If rec!account_curr_bal < Me.eAmount Then
                                err = err + 1
                                MsgBox "The amount you have entered for this payment exceeds the current balance for this account. Please enter a lower payment amount or choose another account.", vbOKOnly, "OVERPAYMENT"
                        End If
                    End If
            
                If err = 0 Then
                
                    sql = "UPDATE payment_info SET PayConsumerNumber = '" & Me.eConsumer & "', PayCheckNum = '" & Me.eCheckNo & "', PayAccountNumber = '" & Me.eAccount & "', PayAmt = " & Me.eAmount & ", PayBucketAmt = " & Me.eAmount & ", PayEnterDate = '" & Format(Me.eDate, "yyyy-MM-dd") & "', PayMemoCode = '" & Me.eMemo & "', PayComment = '" & Me.eComment & "' WHERE PayID = " & Me.ePayID & ";"

                    dB.Execute sql
                    
                End If
            
        End If
        
    End If
        
    DoCmd.OpenForm "sPay_03_check", acNormal
    DoCmd.Close acForm, "sPay_03eCheck", acSaveNo

End Sub
