Private Sub bcbsPostPay_Click()
Dim dB As Database
Dim sql As String
Dim rst As Recordset
    
    If Me.JIRA = "SD-" Or Left(Me.JIRA, 3) <> "SD-" Then
        
            MsgBox "Please enter a JIRA SD Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
            
    Else

    Set dB = CurrentDb
    
        sql = "INSERT INTO payment_info (PayConsumerNumber, PayAccountNumber, PayAmt, PayType, PayLocation, PayEnterDate, PayTenderDate, PayMemoCode, PayInstrument, PayComment, xUser, xType, xComment) "
        sql = sql & "SELECT DM_CONSUMER, DM_ACCOUNT, [Payment Amount], 'PAYMENT', 'CREDITOR', Format([File Date],'yyyy-mm-dd'), Format([File Date],'yyyy-mm-dd'), 'NA', 'NOTAPPLICABLE', BCBS_FILE_NM, '" & Me.usr & "', 'BCP', BCBS_FILE_NM "
        sql = sql & "FROM payBCBS_stg WHERE [Payment Amount] > 0 AND DM_RES = 'READY';"
        dB.Execute (sql)
        
        sql = "INSERT INTO payment_info ( PayConsumerNumber, PayAccountNumber, PayAmt, PayType, PayLocation, PayEnterDate, PayTenderDate, PayMemoCode, PayInstrument, PayComment, xUser, xType, xComment, PayBucketName, PayBucketAmt ) "
        sql = sql & "SELECT DM_CONSUMER, DM_ACCOUNT, [Adjustment Amount], 'ADJUSTMENT', 'CREDITOR', Format([File Date],'yyyy-mm-dd'), Format([File Date],'yyyy-mm-dd'), 'NA', 'NOTAPPLICABLE', BCBS_FILE_NM, '" & Me.usr & " ', 'BCA', BCBS_FILE_NM, 'PRI',[Adjustment Amount] * -1  "
        sql = sql & "FROM payBCBS_stg WHERE [Adjustment Amount] > 0 AND DM_RES = 'READY';"
        dB.Execute (sql)

        Select Case Me.runEnv
        
            Case 1
            
                BCBSpay ("test")
    
                DoCmd.OpenQuery "bcbs_add_qry06_arch", acViewNormal, acEdit

                DoCmd.OpenForm "bcbs_01main", acNormal
                DoCmd.Close acForm, "bcbs_04postPays", acSaveNo
            
            Case 2
            
                BCBSpay ("stage")
    
                DoCmd.OpenQuery "bcbs_add_qry06_arch", acViewNormal, acEdit

                DoCmd.OpenForm "bcbs_01main", acNormal
                DoCmd.Close acForm, "bcbs_04postPays", acSaveNo
            
            Case 3

                DoCmd.OpenForm "bcbs_pst_prod", acNormal
            
        End Select
    
    dB.Close
        
    End If
        
End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close

End Sub
