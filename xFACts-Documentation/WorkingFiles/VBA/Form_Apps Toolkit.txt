VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "Form_Apps Toolkit"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = True
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Private Sub exit_Click()
Dim dB As Database
Dim uName As String
    
    uName = Environ("username")
    
    Set dB = CurrentDb
    dB.Execute ("UPDATE xtsUsers SET lstLogout = Now() WHERE xtsUsername = '" & uName & "';")
    dB.Close
    
    DoCmd.Quit acQuitSaveNone

End Sub
Private Sub Form_Load()
Dim dB As Database
Dim dbz As Long, dbw As String
Dim perm As Recordset, lck As Recordset
Dim uName As String, uRole As String

    Application.SetOption "Auto compact", False
    DoCmd.SetWarnings False
    DoCmd.OpenForm "z_log_scan", , , , , acHidden
    
    uName = Environ("username")
    
    Set dB = CurrentDb
    
    Set lck = dB.OpenRecordset("SELECT * FROM toolMaint")
    Set perm = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & uName & "';")
    
    If lck!maintLocked = -1 Then
    
        DoCmd.SetWarnings True
        
        If perm!xtsRole = "Admin" Then
            DoCmd.OpenForm "z_adm_locked", acNormal
            DoCmd.Close acForm, "Apps Toolkit", acSaveNo
            
        Else
            DoCmd.OpenForm "z_log_locked", acNormal
            DoCmd.Close acForm, "Apps Toolkit", acSaveNo
            
        End If
        
    Else
    
        dbz = FormatNumber(FileLen(CurrentDb.Name) / 1000000, 2)
        dbw = "File Size: " & dbz & " MB"
        
            Select Case dbz
                Case 0 To 75
                    Me.dbSize.ForeColor = RGB(255, 255, 255)
                Case 75.01 To 200
                    Me.dbSize.ForeColor = RGB(255, 153, 0)
                Case Is > 200
                    Me.dbSize.ForeColor = RGB(255, 0, 0)
            End Select
        
        Me.dbSize.Value = dbw
        
        If perm.EOF Then
        
            MsgBox "This application has not been set up for user " & uName & ", please contact the Apps team for access.", vbOKOnly, "ACCESS DENIED"
            DoCmd.Quit acQuitSaveNone
            
        Else
        
            uRole = perm!xtsRole
            Me.xIt.Value = perm!xtsInitials
            
            dB.Execute ("UPDATE xtsUsers SET lstLogin = Now() WHERE xtsUsername = '" & uName & "';")
        
            Select Case perm!xtsRole
                
                Case "Admin"
            
                    Me.gotoAdm.Visible = True
                    Me.dbSize.Visible = True
                    Me.gotoAPIs.Visible = True
                    Me.gotoPays.Enabled = True
                    Me.gotoUpdt.Enabled = True
                
                Case "Apps"
            
                    Me.gotoAdm.Visible = False
                    Me.dbSize.Visible = False
                    Me.gotoAPIs.Visible = True
                    Me.gotoPays.Enabled = True
                    Me.gotoUpdt.Enabled = True
                
                Case "Finance"
            
                    Me.gotoAdm.Visible = False
                    Me.dbSize.Visible = False
                    Me.gotoAPIs.Enabled = False
                    Me.gotoPays.Enabled = True
                    Me.gotoUpdt.Enabled = False
                
                Case "Ops"
            
                    Me.gotoAdm.Visible = False
                    Me.dbSize.Visible = False
                    Me.gotoAPIs.Enabled = False
                    Me.gotoPays.Enabled = False
                    Me.gotoUpdt.Enabled = True
                    
            
                Case "Sales"
            
                    Me.gotoAdm.Visible = False
                    Me.gotoAPIs.Enabled = False
                    Me.gotoPays.Enabled = True
                    Me.gotoUpdt.Enabled = False
            
            End Select
            
        End If
        
    End If

End Sub

Private Sub gotoAdm_Click()
    
    DoCmd.OpenForm "frm_adm_main", acNormal
    DoCmd.Close acForm, "Apps Toolkit", acSaveNo
    
End Sub

Private Sub gotoAPIs_Click()

    DoCmd.OpenForm "Call Rest API", acNormal
    DoCmd.Close acForm, "Apps Toolkit", acSaveNo

End Sub

Private Sub gotoPays_Click()

    DoCmd.OpenForm "fnncl_srvcs", acNormal
    DoCmd.Close acForm, "Apps Toolkit", acSaveNo

End Sub

Private Sub gotoUpdt_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "Apps Toolkit", acSaveNo

End Sub
