Private Sub cbrImpFile_Click()
Dim fs As Object
Dim dB As Database
Dim cnt As Long
Dim fName As String
Dim famt As Recordset


    fName = Me.newFile
    Set fs = CreateObject("Scripting.FileSystemObject")
    err = 0
    
    '   Verify the import filename is valid
        If fs.FileExists(fName) = False Then

                MsgBox "Please enter a valid path & filename for import.", vbCritical, "FILE NOT FOUND"
                
        '  If the file is present, then import (and delete header & trailer records)
            Else
            
            Set dB = CurrentDb
            dB.Execute ("DELETE * FROM CbrOutput_CurrentFile;")
            DoCmd.TransferText acImportFixed, "Metro2_base", "CbrOutput_CurrentFile", fName, False
            dB.Execute ("DELETE * FROM CbrOutput_CurrentFile WHERE Processing_Indicator ='H' OR Processing_Indicator ='T';")
            
                Set famt = dB.OpenRecordset("SELECT * FROM CbrOutput_CurrentFile;")
                famt.MoveLast
                cnt = famt.RecordCount
                
                MsgBox "The specified file has been imported, " & cnt & " records were loaded and are ready for processing.", vbOKOnly, "IMPORT COMPLETE"
    
            dB.Close
            
            DoCmd.OpenForm "cbr_01main", acNormal
            DoCmd.Close acForm, "cbr_02import", acSaveNo
                
        End If
        
End Sub

Private Sub clsCBRimp_Click()

    DoCmd.OpenForm "cbr_01main", acNormal
    DoCmd.Close acForm, "cbr_02import", acSaveNo

End Sub
