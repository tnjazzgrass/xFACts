Private Sub fxdFeeAdd_Click()
Dim dB As Database
Dim amt As Double, pstDue As Double
Dim sql As String, iSQL As String, strSQL As String
Dim rst As Recordset, rs2 As Recordset, rs As ADODB.Recordset

    Set dB = CurrentDb
    
   'See if there are any fees configured for the selected environment that haven't been charged for this period yet
    sql = "SELECT * FROM ff_config WHERE ffcf_fee_active = 'Y' AND ffcf_env = '" & Me.act_env.Value & "' AND ffcf_dm_accnt NOT IN (SELECT cnsmr_accnt_idntfr_agncy_id FROM bdl_fixed_fee WHERE bdl_prd = '" & Me.curr_prd.Value & "' AND bdl_env = '" & Me.act_env.Value & "');"
    Set rst = dB.OpenRecordset(sql)
    
    If rst.EOF Then
    
        MsgBox "There are currently no new fees to process for this period.", vbCritical, "NO FEES CHARGED"
        
    Else

        Do Until rst.EOF
        
           'Figure out the fee amount based on type and rate
            Set rs2 = dB.OpenRecordset("SELECT * FROM ff_type WHERE ffty_fee_cd = '" & rst!ffcf_fee_type & "';")
            
            If rs2!ffty_rate_type = "Percent" Then
                strSQL = "SELECT * FROM vw_crryvr_crdtr_invc WHERE crdtr_id = " & rst!ffcf_crdtr_id & ";"
                    OpenDMrecs rs, strSQL
                amt = ffdf_fee_rate * rs!crdtr_invc_crry_frwrd_bal_amnt
                    rs.Close
                    rs2.Close
            Else
                amt = ffcf_fee_amt
            End If
            
           'Add records to the bdl table for posting
            iSQL = "INSERT INTO bdl_fixed_fee (cnsmr_accnt_idntfr_agncy_id, crdtr_trnsctn_dttm, crdtr_trnsctn_chrg_amnt, crdtr_trnsctn_typ_val_txt, bdl_prd, bdl_env)  "
            iSQL = iSQL & "VALUES ('" & rst!ffcf_dm_accnt & "','" & Now() & "','" & amt & "','SERVICE_EVENT','" & Me.curr_prd.Value & "','" & Me.act_env.Value & "');"
            
            dB.Execute (iSQL)
        
        rst.MoveNext
        Loop
        
        Me.fxd_fee_01_main_sub.Requery
        MsgBox "The active charges for the period of " & Me.curr_prd.Value & " have been added to the " & Me.act_env.Value & " environment queue - please verify the results before posting in DM.", vbInformation, "CHARGES READY FOR POSTING"

    End If
    
    dB.Close

End Sub
