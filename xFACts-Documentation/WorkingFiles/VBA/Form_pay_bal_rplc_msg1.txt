Attribute VB_Name = "Form_pay_bal_rplc_msg1"

Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Compare Database

Private Sub newJIRA_Click()

    Select Case Me.nJIRA.Value
    
        Case "IT-"
        
            MsgBox "Please enter a proper JIRA ticket number to continue.", vbCritical, "INVALID JIRA NUMBER"
            
        Case ""
        
            MsgBox "Please enter a proper JIRA ticket number to continue.", vbCritical, "INVALID JIRA NUMBER"
            
        Case Else
        
            Set [Forms]![pay_bal_rplc_main]![JIRA] = Me.nJIRA.Value
            DoCmd.Close acForm, "pay_bal_rplc_msg1", acSaveNo
            
        End Select
        
End Sub
