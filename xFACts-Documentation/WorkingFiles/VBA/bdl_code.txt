Function PostAccntTags(PATenv As String)
Dim fs, f
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim fld As Field, nfld As Field
Dim cnt As Integer, seq As Integer
Dim LineItem As Recordset, c As Recordset, u As Recordset
Dim BDL_atfile As String, hdr As String, xmlb As String, reg_id As String
Dim fileLoc As String, envURL As String, response As String, jBody As String

    Set dB = CurrentDb
    
    '   Get user ID
        Set u = dB.OpenRecordset("SELECT * FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
    '   Set environment specific variables
        Select Case PATenv
        
            Case "test"
            
                fileLoc = "\\dm-test-app\e$\dmfs\import\bdl\"
                envURL = "http://dm-test-app:8080/dm-rest-services/api"
                
            Case "stage"
            
                fileLoc = "\\dm-stage-app3\dmfs\import\bdl\"
                envURL = "http://dm-stage-app:8080/dm-rest-services/api"
                
            Case "prod"
            
                fileLoc = "\\dm-prod-app3\dmfs\import\bdl\"
                envURL = "http://dm-prod-app:8080/dm-rest-services/api"
                
        End Select
    
   '    Create the empty file for writing to in the location specified above
        BDL_atfile = [Forms]![bdl_pst_accnt_tags]![JIRA] & " Account Tags Batch " & Format(Now(), "yyyyMMdd") & CLng(Timer) & ".txt"
        DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", fileLoc & BDL_atfile

   '    Select the unprocessed records from the VOApps AR table
        Set LineItem = dB.OpenRecordset("SELECT * FROM bdl_accnt_tag WHERE bdl_usr = '" & u!xtsInitials & "' AND bdl_proc = 'N';")
        
        '   Get the count of records to be written from the table
            LineItem.MoveLast
            cnt = LineItem.RecordCount
            LineItem.MoveFirst

   '    Compose the XML file header
        hdr = "<?xml version=""1.0"" encoding=""UTF-8""?>"
        hdr = hdr & Chr(13) & Chr(10) & "<dm_data xmlns=""http://www.fico.com/xml/debtmanager/data/v1_0"">"
        hdr = hdr & Chr(13) & Chr(10) & "  <header>"
        hdr = hdr & Chr(13) & Chr(10) & "    <sender_id_txt>Organization</sender_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "    <target_id_txt>FAC Debt Manager</target_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "    <batch_id_txt>" & Left(BDL_atfile, 5) & "</batch_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "      <operational_transaction_type>CONSUMERACCOUNT</operational_transaction_type>"
        hdr = hdr & Chr(13) & Chr(10) & "         <total_count>" & cnt & "</total_count>"
        hdr = hdr & Chr(13) & Chr(10) & "         <creation_data>" & Format(Date, "yyyy-mm-dd") & "T" & Format(Time(), "Short time") & ":00</creation_data>"
        hdr = hdr & Chr(13) & Chr(10) & "      <custom_properties>"
        hdr = hdr & Chr(13) & Chr(10) & "        <custom_property/>"
        hdr = hdr & Chr(13) & Chr(10) & "      </custom_properties>"
        hdr = hdr & Chr(13) & Chr(10) & "  </header>"
        hdr = hdr & Chr(13) & Chr(10) & "  <operational_transaction_data>"
        hdr = hdr & Chr(13) & Chr(10) & "    <account_operational_transaction_data>"
    
    '   Write the header to file
        Set fs = CreateObject("Scripting.FileSystemObject")
        Set f = fs.OpenTextFile(fileLoc & BDL_atfile, ForAppending, TristateFalse)
        
        f.Write hdr
        
    '   Begin composing the AR Event XML records
        seq = 1
        
            Do Until LineItem.EOF
            
                xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "<cnsmr_accnt_tag seq_no=""" & seq & """ type=""CONSUMER_ACCOUNT_TAG"">"
                xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "<cnsmr_accnt_idntfr_agncy_id>" & LineItem!cnsmr_accnt_idntfr_agncy_id & "</cnsmr_accnt_idntfr_agncy_id>"
                xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "<tag_shrt_nm>" & LineItem!tag_shrt_nm & "</tag_shrt_nm>"
                xmlb = xmlb & IIf(Len(LineItem!cnsmr_accnt_tag_assgn_dt > 0), Chr(13) & Chr(10) & Space(8) & "<cnsmr_accnt_tag_assgn_dt>" & LineItem!cnsmr_accnt_tag_assgn_dt & "</cnsmr_accnt_tag_assgn_dt>", "")
                xmlb = xmlb & IIf(Len(LineItem!cnsmr_accnt_tag_ordr_nmbr > 0), Chr(13) & Chr(10) & Space(8) & "<cnsmr_accnt_tag_ordr_nmbr>" & LineItem!cnsmr_accnt_tag_ordr_nmbr & "</cnsmr_accnt_tag_ordr_nmbr>", "")
                xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "</cnsmr_accnt_tag>"
            
            f.Write xmlb
            xmlb = ""
            seq = seq + 1
            
            LineItem.MoveNext
            
            Loop
     '  CLose the last XML Nodes and write to file
        f.Write Chr(13) & Chr(10) & "    </account_operational_transaction_data>"
        f.Write Chr(13) & Chr(10) & "  </operational_transaction_data>"
        f.Write Chr(13) & Chr(10) & "</dm_data>"
        
        f.Close
             
'   Register the BDL file

    Set c = dB.OpenRecordset("SELECT * FROM APIDeets;")

    jBody = "{"
    jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileName"":""" & BDL_atfile & ""","
    jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileType"":""BDL_IMPORT"""
    jBody = jBody & Chr(13) & Chr(10) & "}"

        httpReq.Open "POST", envURL & "/fileregistry", False, c!uName, c!pwd
        httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
        httpReq.setRequestHeader "Authorization", c!auth
        httpReq.sEnd (jBody)

        response = httpReq.responseText
        reg_id = Mid(response, 46, 6)
        
    jBody = ""

'    Debug.Print response

'   Import the registered BDL file
    httpReq.Open "POST", envURL & "/fileregistry/" & reg_id & "/bdlimport", False, c!uName, c!pwd
    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
    httpReq.setRequestHeader "Authorization", c!auth
    httpReq.sEnd (jBody)
    
    response = httpReq.responseText
'    Debug.Print Chr(13) & Chr(10) & response
        
'   Mark all the records as processed
    dB.Execute ("UPDATE bdl_accnt_tag SET bdl_proc = 'Y', bdl_post = NOW(), bdl_env = '" & PATenv & "' WHERE bdl_proc = 'N' AND bdl_usr = '" & u!xtsInitials & "';")
    dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('postAccntTags', '" & PATenv & "', '" & Environ("username") & "', Now())"
    dB.Close
        
    MsgBox "BDL file " & BDL_atfile & " has been submitted in Debt Manager. Check the Consumer/Account Data Processing Screen in DM to verify that it processed completely.", vbInformation, "BDL FILE SUBMITTED"

End Function

Function RemoveMailReturns(MRenv As String)
Dim del As Long
Dim mrdb As Database
Dim httpReq As New XMLHTTP60
Dim rst1 As Recordset, c As Recordset, x As Recordset, rs As ADODB.Recordset
Dim response As String, jBody As String, sql As String, sURL As String, strSQL As String

    Set mrdb = CurrentDb
    DoCmd.OpenForm "bdl_mr_count", acNormal
    
'   Set environment specific variables
    Select Case MRenv
    
        Case "test"
        
            sURL = "http://dm-test-app:8080/dm-rest-services/api"
            
        Case "stage"
        
            sURL = "http://dm-stage-app:8080/dm-rest-services/api"
            
        Case "prod"
        
            sURL = "http://dm-prod-app:8080/dm-rest-services/api"
            
    End Select

    Set rst1 = mrdb.OpenRecordset("SELECT * FROM bdl_addrss_updt WHERE bdl_usr = '" & [Forms]![bdl_pst_addr_updts]![usr] & "' AND bdl_proc = 'N' ORDER BY cnsmr_idntfr_agncy_id;")
    Set c = mrdb.OpenRecordset("SELECT * FROM APIDeets;")

    Do Until rst1.EOF
    
        '   Set the ADODB sql string
            strSQL = "SELECT cnsmr_addrss_mail_rtrn_cd AS mr FROM crs5_oltp.dbo.cnsmr_addrss WHERE cnsmr_id = (SELECT cnsmr_id FROM crs5_oltp.dbo.cnsmr WHERE cnsmr_idntfr_agncy_id = 56416611);"

        '   Find out if there's a mail return
            Select Case MRenv
            
                Case "test"
                
                    OpenDMrect rs, strSQL
                    
                Case "stage"
                
                    OpenDMrecg rs, strSQL
                    
                Case "prod"
                
                    OpenDMrecs rs, strSQL
                    
            End Select
            
            If IsNull(rs!mr) Then
            
                response = "No mail return present"
                
            Else
        
                httpReq.Open "DELETE", sURL & "/consumers/" & rst1!cnsmr_idntfr_agncy_id & "/addressmailreturn", False, c!uName, c!pwd
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", c!auth
                httpReq.sEnd
        
                response = httpReq.responseText
                
            End If
        
            sql = "UPDATE bdl_addrss_updt SET bdl_mr_resp = '" & response & "', bdl_mr_proc = 'Y' WHERE cnsmr_idntfr_agncy_id = '" & rst1!cnsmr_idntfr_agncy_id & "' AND bdl_usr = '" & [Forms]![bdl_pst_addr_updts]![usr] & "' AND bdl_proc = 'N';"
            mrdb.Execute (sql)
            
            del = Timer + 1
            Do While Timer < del
                Set x = mrdb.OpenRecordset("SELECT COUNT(*) AS cnt FROM bdl_addrss_updt WHERE bdl_usr = '" & [Forms]![bdl_pst_addr_updts]![usr] & "' AND bdl_mr_proc = 'Y' AND bdl_proc = 'N';")
                    Forms("bdl_mr_count").Controls("apiComp").Value = x!cnt
                Set x = Nothing
                DoEvents
            Loop
    
    rst1.MoveNext
    Loop
    
    DoCmd.Close acForm, "bdl_mr_count", acSaveNo
    mrdb.Close

    MsgBox "The API calls to delete mail returns have been submitted", vbInformation, "MAIL RETURN APIs SUBMITTED"

End Function

Function ImportNB(nbEnv As String)
Dim fs, f
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim fld As Field, nfld As Field, err As Long, rb As String
Dim cnt As Integer, seq As Integer, bsn As String, ss As Integer
Dim XML_nbfile As String, hdr As String, xmlb As String, reg_id As String
Dim con As Recordset, acc As Recordset, c As Recordset, u As Recordset, rs As ADODB.Recordset
Dim fileLoc As String, envURL As String, response As String, jBody As String, strSQL As String

    Set dB = CurrentDb
    
    '   Get user ID
        Set u = dB.OpenRecordset("SELECT * FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
    '   Set environment specific variables
        Select Case nbEnv
        
            Case "test"
            
                fileLoc = "\\dm-test-app\e$\dmfs\import\bdl\"
                envURL = "http://dm-test-app:8080/dm-rest-services/api"
                
            Case "stage"
            
                fileLoc = "\\dm-stage-app3\dmfs\import\bdl\"
                envURL = "http://dm-stage-app:8080/dm-rest-services/api"
                
            Case "prod"
            
                fileLoc = "\\dm-prod-app3\dmfs\import\bdl\"
                envURL = "http://dm-prod-app:8080/dm-rest-services/api"
                
        End Select
        
   '    Create unique new biz batch short name
        err = 1
        Do Until err = 0
        
            cnt = 0
            rb = "ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890"
            Do Until cnt = 8
                ss = RandBetween(1, 33)
                bsn = bsn & Mid(rb, ss, 1)
                cnt = cnt + 1
            Loop
            
            Set strSQL = "SELECT COUNT(*) AS nc FROM new_bsnss_btch WHERE new_bsnss_btch_shrt_nm = '" & bsn & "';"
                
                Select Case nbEnv
                
                    Case "test"
                        OpenDMrect rs, strSQL
                        
                    Case "stage"
                        OpenDMrecg rs, strSQL
                        
                    Case "prod"
                        OpenDMrecs rs, strSQL
                        
                End Select
                
            err = rs!nc
            strSQL = ""
            rs.Close
            
        Loop
    
   '    Create the empty file for writing to in the location specified above
        XML_nbfile = "NB_SAMPLE_RECS " & Format(Now(), "yyyyMMdd") & CLng(Timer) & ".txt"
        DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", fileLoc & XML_nbfile

   '    Select the consumer detail records from the test record table
        Set c = dB.OpenRecordset("SELECT DISTINCT ConsumerLegacyID FROM nb_test_records ORDER BY ConsumerLegacyID;")
        
        '   Get the count of records to be written from the table
            c.MoveLast
            cnt = LineItem.RecordCount
            c.MoveFirst

    '   Start with the XML header (this routine just sets all the counts to 1 since they're not used by the system anyway)
        hdr = "<?xml version=""1.0"" encoding=""UTF-8""?><newbiz xmlns=""http://www.crsoftwareinc.com/xml/ns/titanium/common/v1_0"" xmlns:xsi=""http://www.w3.org/2001/XMLSchema-instance"">"
        hdr = hdr & Chr(13) & Chr(10) & "  <newbiz-header>"
        hdr = hdr & Chr(13) & Chr(10) & "     <total-consumers>1</total-consumers>"
        hdr = hdr & Chr(13) & Chr(10) & "     <creditor-name></creditor-name>"
        hdr = hdr & Chr(13) & Chr(10) & "     <total-principal>1</total-principal>"
        hdr = hdr & Chr(13) & Chr(10) & "     <total-charge>1</total-charge>"
        hdr = hdr & Chr(13) & Chr(10) & "     <total-interest>1</total-interest>    <total-balance>1</total-balance>"
        hdr = hdr & Chr(13) & Chr(10) & "  </newbiz-header>"
        hdr = hdr & Chr(13) & Chr(10) & "  <consumers>"
    
    '   Write the header to file
        Set fs = CreateObject("Scripting.FileSystemObject")
        Set f = fs.OpenTextFile(fileLoc & NB_actfile, ForAppending, TristateFalse)
        
        f.Write hdr
        
    '   Begin composing the AR Event XML records
        seq = 1
        
        Do Until c.EOF
        
        Set con = dB.OpenRecordset("SELECT TOP 1 * FROM nb_test_records WHERE ConsumerLegacyID = '" & c!ConsumerLegacyID & "';")
        
        '   Compose the XML for the basic consumer node
            xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "<consumer>"

            xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<consumer-name>"

            xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "</consumer-name>"

            'xmlb = xmlb & IIf(LineItem![Commercial Contact Name] <> "", Chr(13) & Chr(10) & Space(10) & "<commercial-contact-name>" & LineItem![Commercial Contact Name] & "</commercial-contact-name>", "")
            'xmlb = xmlb & IIf(LineItem![Commerical Consumer] = "Y", Chr(13) & Chr(10) & Space(10) & "<commercial-name>" & LineItem![Commerical Name] & "</commercial-name>", "")
            xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "</consumer-info>"

                    xmlb = xmlb & Chr(13) & Chr(10) & Space(12) & ca

            xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "</consumer-address-infos>"
