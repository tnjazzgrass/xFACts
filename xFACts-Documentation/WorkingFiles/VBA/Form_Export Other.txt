Private Sub AcctTags_Click()
Dim fs, f
Dim fld As Field, nfld As Field
Dim cnt As Integer, seq As Integer
Dim tdf As TableDef, ntdf As TableDef
Dim LineItem As Recordset, c As Recordset
Dim BDL_Acctfile As String, hdr As String, xmlb As String, nullify As String

    Set dB = CurrentDb

dB.Execute "UPDATE FileCnt SET FileCnt.flcnt = [flcnt]+1;"
Set c = dB.OpenRecordset("SELECT flcnt FROM FileCnt;")
BDL_Acctfile = Me.outDir & "BDL_cnsmr_accnt_tag_" & c!flcnt & ".txt"
DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", BDL_Acctfile

    Set LineItem = dB.OpenRecordset("SELECT * FROM cnsmr_accnt_tag;")
    
        LineItem.MoveLast
        cnt = LineItem.RecordCount
        LineItem.MoveFirst

        hdr = "<?xml version='1.0' encoding='UTF-8'?>"
        hdr = hdr & Chr(13) & Chr(10) & "<dm_data xmlns=""http://www.fico.com/xml/debtmanager/data/v1_0"">"
        hdr = hdr & Chr(13) & Chr(10) & "  <header>"
        hdr = hdr & Chr(13) & Chr(10) & "    <sender_id_txt>Organization</sender_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "    <target_id_txt>target_id_1</target_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "    <batch_id_txt>B" & c!flcnt & "</batch_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "    <operational_transaction_type>CONSUMERACCOUNT</operational_transaction_type>"
        hdr = hdr & Chr(13) & Chr(10) & "    <total_count>" & cnt & "</total_count>"
        hdr = hdr & Chr(13) & Chr(10) & "    <creation_data>" & Format(Date, "yyyy-mm-dd") & "T" & Format(Time(), "Short time") & ":00.001-05:00</creation_data>"
        hdr = hdr & Chr(13) & Chr(10) & "    <custom_properties>"
        hdr = hdr & Chr(13) & Chr(10) & "      <custom_property/>"
        hdr = hdr & Chr(13) & Chr(10) & "    </custom_properties>"
        hdr = hdr & Chr(13) & Chr(10) & "  </header>"
        hdr = hdr & Chr(13) & Chr(10) & "  <operational_transaction_data>"
        hdr = hdr & Chr(13) & Chr(10) & "    <account_operational_transaction_data>"

    Set fs = CreateObject("Scripting.FileSystemObject")
    Set f = fs.OpenTextFile(BDL_Acctfile, ForAppending, TristateFalse)
    
    f.Write hdr
    
    seq = 1
    nullify = ""
    
        Do Until LineItem.EOF
        
            xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "<cnsmr_accnt_tag seq_no=""" & seq & """ type=""CONSUMER_ACCOUNT_TAG"">"

                    Set ntdf = dB.TableDefs("cnsmr_accnt_tag")
                    
                        For Each nfld In ntdf.Fields
                                
                            nullify = nullify & IIf(LineItem(nfld.Name) = "NULL", Chr(13) & Chr(10) & Space(10) & "<nullify_field>" & nfld.Name & "</nullify_field>", "")
                            
                        Next
                        
                    If Len(nullify) > 0 Then
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "<nullify_fields>" & nullify & Chr(13) & Chr(10) & Space(8) & "</nullify_fields>"
                    End If
            
                Set tdf = dB.TableDefs("cnsmr_accnt_tag")
                
                    For Each fld In tdf.Fields
                            
                        xmlb = xmlb & IIf(IsNull(LineItem(fld.Name)), "", IIf(LineItem(fld.Name) = "NULL", "", Chr(13) & Chr(10) & Space(8) & "<" & fld.Name & ">" & LineItem(fld.Name) & "</" & fld.Name & ">"))
                        
                    Next
                
            xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "</cnsmr_accnt_tag>"
        
        f.Write xmlb
        xmlb = ""
        nullify = ""
        seq = seq + 1
        
        LineItem.MoveNext
        
        Loop
        
            f.Write Chr(13) & Chr(10) & "    </account_operational_transaction_data>"
            f.Write Chr(13) & Chr(10) & "  </operational_transaction_data>"
            f.Write Chr(13) & Chr(10) & "</dm_data>"
            
            f.Close
            dB.Close

x = MsgBox("Export complete, ready for BDL processing", vbOKOnly, "FILE READY")
End Sub

        Do Until actPt.EOF

            Me.Recalc
            Me.Repaint
            
            Set rst = wDb.OpenRecordset("SELECT RecordType, TotalCnt, RecLen FROM RecordCnt WHERE TotalCnt>0 AND RecordType < 'Record35' OR TotalCnt>0 AND RecordType > 'Record59';")
        
            Do Until rst.EOF
            
                dLine = ""
                
                rType = IIf(rst!RecordType = "Record37", " (Primary)", "")
                sql = "SELECT * FROM " & rst!RecordType & " WHERE [Unique Consumer ID" & rType & "]='" & actPt![Unique Consumer ID] & "';"
                Set rsChk = wDb.OpenRecordset(sql)
                
                While Not rsChk.EOF
                        
                    Set tdf = wDb.TableDefs(rst!RecordType)
                    
                        For Each fld In tdf.Fields
                        
                                If fld.Type <> 10 Then
                                    fldName = fld.Name & tdf.Name
                                    Select Case fldName
                                        Case "CommentRecord04", "Creditor DescriptionRecord60"
                                            fldSize = 512
                                        Case "CommentRecord80", "Custom Page Information FieldRecord15", "Custom Page Information FieldRecord42", "Custom Page Information FieldRecord74"
                                            fldSize = 500
                                        Case "RemarksRecord50", "DescriptionRecord52", "Supplemental InformationRecord52", "Reason Not CoveredRecord54"
                                            fldSize = 1024
                                        Case Else
                                            fldSize = 256
                                    End Select
                                Else
                                    fldSize = fld.Size
                                End If
                            
                            dLine = dLine & rsChk(fld.Name) & Space(fldSize - IIf(IsNull(rsChk(fld.Name)), 0, Len(rsChk(fld.Name))))
                            
                            If rst!RecordType = "Record31" Then
                                ConID = rsChk![Unique Consumer ID]
                                AccID = rsChk![Unique Consumer Account ID]
                            End If

                        Next
                
                    dLine = dLine & Chr(13) & Chr(10)
                    
                    If rst!RecordType = "Record31" Then
                        dLine = dLine & MedRec(ConID, AccID)
                    End If
                
                    If dLine <> "" Then
                        f.Write dLine
                        dLine = ""
                    End If
                    
                    rsChk.MoveNext
                    
                    Wend
            
            rst.MoveNext
            Loop

        actPt.MoveNext
        Loop
        
    f.Close
    wDb.Close

    Set oFS = New FileSystemObject
    Set oFS1 = New FileSystemObject
    Set oTS = oFS.OpenTextFile("C:\TestData\Record99.txt", ForReading)
    vTemp = oTS.ReadAll
    oTS.Close
    
    Set oTS1 = oFS.OpenTextFile(FCL, ForAppending, True)
    oTS1.Write (vTemp)
    oTS1.Close
    
    Kill "C:\TestData\Record99.txt"
    
    x = MsgBox("Export complete, ready for XML encoding", vbOKOnly, "FILE READY")
    
End Sub
