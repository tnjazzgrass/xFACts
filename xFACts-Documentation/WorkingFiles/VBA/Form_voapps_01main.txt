Private Sub opnPostVOApps_Click()
Dim dB As Database
Dim rec As Recordset

    Set dB = CurrentDb
    Set rec = dB.OpenRecordset("SELECT * FROM voapps_stg WHERE vstg_proc='N';")
        If rec.EOF Then
            MsgBox "Please import new records to post before continuing with the process.", vbOKOnly, "NO NEW EVENTS TO POST"
        Else
            DoCmd.OpenForm "voapps_04postARs", acNormal
            DoCmd.Close acForm, "voapps_01main", acSaveNo
        End If
    dB.Close
    
End Sub
