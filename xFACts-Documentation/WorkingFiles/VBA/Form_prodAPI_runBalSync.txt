Option Compare Database

Private Sub balProdSync_Click()
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String
Dim sURL As String, sUser As String, sPass As String, sAuth As String

    Set dB = CurrentDb
    Set rst1 = dB.OpenRecordset("SELECT MAX(toolDate) AS lst FROM toolUse WHERE toolFunction = 'balSync' AND toolEnv = 'PROD';")
        
        If DateDiff("n", rst1!lst, Now()) < 60 Then
        
            MsgBox "This API was called less than one hour ago, please wait until the last run has completed", vbInformation, "API RECENTLY CALLED"
            DoCmd.Close acForm, "prodAPI_runBalSync", acSaveNo
            
        Else
        
            sURL = "http://dm-prod-app:8080/dm-rest-services/api/scheduledjobs/UPDATE_BALANCES"
             
                httpReq.Open "POST", sURL, False, [Forms]![Call REST API]![APIuser], [Forms]![Call REST API]![APIpass]
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", [Forms]![Call REST API]![authCode]
                httpReq.sEnd
                
                response = httpReq.responseText
               
            dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('balSync', 'PROD', '" & Environ("username") & "', Now())"
            DoCmd.Close acForm, "prodAPI_runBalSync", acSaveNo
            
            MsgBox response, vbInformation, "PRODUCTION WEBSERVICE RESPONSE"
        
            [Forms]![Call REST API].Visible = True
            
        End If

End Sub
