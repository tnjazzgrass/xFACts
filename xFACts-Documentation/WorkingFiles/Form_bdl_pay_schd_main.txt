Private Sub backPaySched_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pay_schd_main", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.bdl_pay_schd_sum.Requery
    Me.bdl_pay_schd_ins.Requery
    Me.bdl_pay_schd_asc.Requery
    
End Sub

Private Sub schedPays_Click()
Dim dB As Database
Dim err As Integer
Dim sqli As String, sqla As String
Dim chks As Recordset, chki As Recordset, chka As Recordset

    err = 0

    If Me.JIRA = "IT-" Or Left(Me.JIRA, 3) <> "IT-" Then
        
        MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
        err = err + 1
            
    End If
    
    If err = 0 Then

        Set dB = CurrentDb
        
            sqli = "SELECT Count(bdl_pay_schd_smy.schdld_pymnt_smmry_idntfr) AS inst "
            sqli = sqli & "FROM bdl_pay_schd_smy LEFT JOIN bdl_pay_schd_ins ON bdl_pay_schd_smy.schdld_pymnt_smmry_idntfr = bdl_pay_schd_ins.schdld_pymnt_smmry_idntfr "
            sqli = sqli & "WHERE (((bdl_pay_schd_ins.schdld_pymnt_smmry_idntfr) Is Null) AND ((bdl_pay_schd_ins.bdl_proc)='N') AND ((bdl_pay_schd_ins.bdl_usr)='" & [Forms]![bdl_pay_schd_main]![usr] & "'));"
            
            sqla = "SELECT Count(bdl_pay_schd_smy.schdld_pymnt_smmry_idntfr) AS accts "
            sqla = sqla & "FROM bdl_pay_schd_smy LEFT JOIN bdl_pay_schd_asc ON bdl_pay_schd_smy.schdld_pymnt_smmry_idntfr = bdl_pay_schd_asc.schdld_pymnt_smmry_idntfr "
            sqla = sqla & "WHERE (((bdl_pay_schd_asc.schdld_pymnt_smmry_idntfr) Is Null) AND ((bdl_pay_schd_asc.bdl_proc)='N') AND ((bdl_pay_schd_asc.bdl_usr)='" & [Forms]![bdl_pay_schd_main]![usr] & "'));"
        
        Set chks = dB.OpenRecordset("SELECT * FROM bdl_pay_schd_smy WHERE bdl_proc = 'N' AND bdl_usr = '" & Me.usr.Value & "';")
        Set chki = dB.OpenRecordset(sqli)
        Set chka = dB.OpenRecordset(sqla)
        
        If chks.EOF Then
        
            MsgBox "Please enter accounts before attempting to run the process.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
            err = err + 1
            
        End If
        
        If chki!inst > 0 Then
        
            MsgBox "There are summaries that are missing installments - please correct the data and try again.", vbCritical, "MISSING INSTALLMENTS"
            err = err + 1
            
        End If
        
        If chka!accts > 0 Then
        
            MsgBox "There are summaries that don't have any accounts associated with them - please correct the data and try again.", vbCritical, "MISSING ACCOUNTS"
            err = err + 1
            
        End If
        
        dB.Close
        
        If err = 0 Then
                
            Select Case Me.runEnv
        
                Case 1
                
                    ImportPaySchedules ("test")
            
                Case 2
                
                    ImportPaySchedules ("stage")
            
                Case 3
                
                    ImportPaySchedules ("prod")
                    
            End Select
                
        End If
            
    End If

End Sub
