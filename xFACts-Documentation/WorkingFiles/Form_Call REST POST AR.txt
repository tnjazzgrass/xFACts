Private Sub APImoveAcct_Click()
Dim del As Long
Dim dB As Database
Dim rst1 As Recordset
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String

    Set dB = CurrentDb
    Set rst1 = dB.OpenRecordset("SELECT * FROM rData WHERE Processed = 'N'")

        jBody = "{"
        jBody = jBody & Chr(13) & Chr(10) & """consumerAgencyId"":" & rst1!ConsumerID & ","

            httpReq.setRequestHeader "Authorization", Me.authCode

Private Sub sendBBalCorr_Click()
Dim del As Long
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim rst As Recordset, rst1 As Recordset, rst2 As Recordset, rst3 As Recordset
Dim response As String, jBody As String, sql As String, astr As String, dt As String

    Set dB = CurrentDb
    dB.Execute ("DELETE FROM bal_corr_consumers_string")
    
    Set rst = dB.OpenRecordset("SELECT unix_date FROM unix_dates WHERE Today = Date()")
    dt = rst!unix_date
    
    Set rst1 = dB.OpenRecordset("SELECT DISTINCT ConsumerID FROM bal_corr_accounts WHERE Processed = 'N'")
    
    If rst1.EOF Then
    
        MsgBox "Please populate the bal_corr_accounts table.", vbInformation, "NO RECORDS FOUND"
        
    Else
    
        dB.Execute ("INSERT INTO bal_corr_consumers_string ( ConsumerID ) SELECT DISTINCT ConsumerID FROM bal_corr_accounts WHERE Processed = 'N' GROUP BY ConsumerID")
        
        Do Until rst1.EOF
        
            astr = ""
            
            sql = "SELECT * FROM bal_corr_accounts WHERE Processed = 'N' AND ConsumerID = '" & rst1!ConsumerID & "' ORDER BY ConsumerID, AccountIDs"
            Set rst2 = dB.OpenRecordset(sql)
            
                Do Until rst2.EOF
        
                    astr = astr & "," & rst2!AccountIDs

                rst2.MoveNext
                Loop
                
            dB.Execute ("UPDATE bal_corr_accounts SET Processed = 'Y' WHERE ConsumerID = '" & rst1!ConsumerID & "'")
            dB.Execute ("UPDATE bal_corr_consumers_string SET AccountString = '" & Mid(astr, 2, Len(astr)) & "' WHERE ConsumerID = '" & rst1!ConsumerID & "'")
            
        rst1.MoveNext
        Loop
        
        Set rst3 = dB.OpenRecordset("SELECT * FROM bal_corr_consumers_string WHERE ResponseStr IS NULL")
        Do Until rst3.EOF
                    
            jBody = "{"
            jBody = jBody & Chr(13) & Chr(10) & """onDemand"": false,"
            jBody = jBody & Chr(13) & Chr(10) & """documentTemplateShortName"": ""BBalCorr"","
            jBody = jBody & Chr(13) & Chr(10) & """deliveryChannelTypeReference"": ""PRINT"","
            jBody = jBody & Chr(13) & Chr(10) & """sendDate"": " & dt & ","
            jBody = jBody & Chr(13) & Chr(10) & """sendToEntityDescriptors"": ["
            jBody = jBody & Chr(13) & Chr(10) & "{"
            jBody = jBody & Chr(13) & Chr(10) & """entityIdentifier"": " & rst3!ConsumerID & ","
            jBody = jBody & Chr(13) & Chr(10) & """entityType"": ""CONSUMER"""
            jBody = jBody & Chr(13) & Chr(10) & "}"
            jBody = jBody & Chr(13) & Chr(10) & "],"
            jBody = jBody & Chr(13) & Chr(10) & """accountsInNotice"": ["
            jBody = jBody & Chr(13) & Chr(10) & rst3!AccountString
            jBody = jBody & Chr(13) & Chr(10) & "]"
            jBody = jBody & Chr(13) & Chr(10) & "}"
        
                httpReq.Open "POST", Me.APIurl & "consumers/" & rst3!ConsumerID & "/noticerequests", False, Me.APIuser, Me.APIpass
