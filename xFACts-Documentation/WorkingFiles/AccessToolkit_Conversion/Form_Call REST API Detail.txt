Private Sub exit_Click()

    DoCmd.OpenForm "Call REST API", acNormal
    DoCmd.Close acForm, "Call REST API Detail", acSaveNo
    
End Sub
Private Sub post_api_Click()
Dim sURL As String
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, aurl As String

    Select Case Me.runEnv
    
        Case 1
            sURL = "http://DM-TEST-APP/dm-rest-services/api/"
        
        Case 2
            sURL = "http://stage-app.fac.local:8080/dm-rest-services/api/"
        
        Case 3
            sURL = "http://DM-PROD-APP3/dm-rest-services/api/"
            
    End Select

    jBody = Me.api_body
    
    Debug.Print jBody
    
    httpReq.Open Me.api_method, sURL & Me.api_url, False, Me.APIuser, Me.APIpass
    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
    httpReq.setRequestHeader "Authorization", Me.authCode
    httpReq.sEnd (jBody)
        
    response = httpReq.responseText
    MsgBox response, vbInformation, "API CALL SUBMITTED"

End Sub
