Function PostAdjustments(PayEnv As String)
Dim fs, f
Dim dB As Database
Dim httpReq As New XMLHTTP60, awt As Long, del As Long
Dim cnt As Integer, seq As Integer, cpj As Integer, envURL As String, pst As Long
Dim response As String, jBody As String, reg_id As String, cpb_id As String, fileLoc As String
Dim pay_batch As String, pay_file As String, hdr As String, xmlb As String, axml As String, strSQL As String, opt As String
Dim b As Recordset, c As Recordset, u As Recordset, cpy As Recordset, apy As Recordset, api As Recordset, rs As ADODB.Recordset, fin As Recordset

    Set dB = CurrentDb
    
    '   Set environment specific variables
        Select Case PayEnv
        
            Case "test"
            
                fileLoc = "\\dm-test-app\e$\dmfs\import\payments\"
                envURL = "http://dm-test-app:8080/dm-rest-services/api"
                
            Case "stage"
            
                fileLoc = "\\dm-stage-app3\dmfs\import\payments\"
                envURL = "http://dm-stage-app:8080/dm-rest-services/api"
                
            Case "prod"
            
                fileLoc = "\\dm-prod-app3\dmfs\import\payments\"
                envURL = "http://dm-prod-app:8080/dm-rest-services/api"
                
        End Select
        
    Set api = dB.OpenRecordset("SELECT * FROM xtsCred;")
    Set u = dB.OpenRecordset("SELECT * FROM toolUsers WHERE toolUsername = '" & Environ("username") & "';")
        
    If [Forms]![adj_batch_01_main]![eligibleOver] = -1 Then
        opt = Chr(13) & Chr(10) & "  <payment-batch-options>"
        opt = opt & Chr(13) & Chr(10) & "    <acceptable-failure-rate>1.00</acceptable-failure-rate>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-payment-adjustment-eligibility-check>true</skip-payment-adjustment-eligibility-check>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-forgive-balance-adjustment>true</skip-forgive-balance-adjustment>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-financial-transaction-ar-event>true</skip-financial-transaction-ar-event>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-post-financial-transaction-rule>true</skip-post-financial-transaction-rule>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-compliance-overpayment-rule>true</skip-compliance-overpayment-rule>"
        opt = opt & Chr(13) & Chr(10) & "    <skip-credit-bureau-updates>true</skip-credit-bureau-updates>"
        opt = opt & Chr(13) & Chr(10) & "  </payment-batch-options>"
    
        Else
            opt = ""
    
    End If
       
    dB.Execute "UPDATE payment_info INNER JOIN payMemos ON payment_info.PayMemoCode = payMemos.memo_cd_shrt_nm SET payment_info.PayInstrument = [pymnt_instrmnt_typ_cd] WHERE (((payment_info.PayInstrument) Is Null));"
    dB.Execute "UPDATE payment_info SET PayBucketAmt = PayAmt * -1 WHERE xProc = 'N' AND xType = 'ADJ' AND xUser = '" & u!toolInitials & "' AND xComment IS NULL;"
    
    pay_batch = [Forms]![adj_batch_01_main]![JIRA] & "_FINANCIAL_BATCH_" & Format(Now(), "yyyyMMdd") & CLng(Timer)
    pay_file = pay_batch & ".txt"
        
        Set b = dB.OpenRecordset("SELECT SUM(PayAmt) AS payTot FROM payment_info WHERE xProc = 'N' AND xType = 'ADJ' AND xUser = '" & u!toolInitials & "' AND xComment IS NULL;")
        Set c = dB.OpenRecordset("SELECT DISTINCT PayConsumerNumber FROM payment_info WHERE xProc = 'N' AND xType = 'ADJ' AND xUser = '" & u!toolInitials & "' AND xComment IS NULL;")
    
            c.MoveLast
            cpj = c.RecordCount
            c.MoveFirst
    
        DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", fileLoc & pay_file
        
        hdr = "<?xml version=""1.0"" encoding=""UTF-8"" standalone=""yes"" ?>"
            hdr = hdr & Chr(13) & Chr(10) & "<consumer-payment-import-job xmlns='http://www.crsoftwareinc.com/xml/ns/titanium/common/v1_0' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>"
            hdr = hdr & Chr(13) & Chr(10) & "  <payment-import-header>"
            hdr = hdr & Chr(13) & Chr(10) & "     <batch-code>BATCH</batch-code>"
            hdr = hdr & Chr(13) & Chr(10) & "     <batch-name>" & [Forms]![adj_batch_01_main]![JIRA] & " ADJUSTMENTS</batch-name>"
            hdr = hdr & Chr(13) & Chr(10) & "     <number-of-accounts>" & cpj & "</number-of-accounts>"
            hdr = hdr & Chr(13) & Chr(10) & "     <total-value-of-account-payment>" & b!payTot & "</total-value-of-account-payment>"
            hdr = hdr & Chr(13) & Chr(10) & "     <date-of-turnover>" & Format(Date, "yyyy-mm-dd") & "</date-of-turnover>"
            hdr = hdr & Chr(13) & Chr(10) & "     <is-conversion-job>false</is-conversion-job>"
            hdr = hdr & Chr(13) & Chr(10) & "  </payment-import-header>"
            hdr = hdr & IIf([Forms]![adj_batch_01_main]![eligibleOver] = -1, opt, "")
            hdr = hdr & Chr(13) & Chr(10) & "  <consumer-payment-imports>"
    
         Set fs = CreateObject("Scripting.FileSystemObject")
        Set f = fs.OpenTextFile(fileLoc & pay_file, ForAppending, TristateFalse)
        
        f.Write hdr
        
        seq = 1
        
        '   Compile Adjs
        '   Find all the cpj's first
            Set cpy = dB.OpenRecordset("SELECT DISTINCT PayConsumerNumber, PayType, First(PayComment) AS PC, Sum(PayAmt) AS PD, Min(PayEnterDate) AS ED, First(PayLocation) AS PL, First(PayMemoCode) AS MC, First(PayInstrument) AS PI FROM payment_info WHERE xProc = 'N' AND xUser = '" & u!toolInitials & "' AND xType = 'ADJ' GROUP BY PayConsumerNumber, PayType ORDER BY PayConsumerNumber;")
            
                Do Until cpy.EOF
                
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "<consumer-payment-import>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "<consumer-payment>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<import-record-identifier>" & seq & "</import-record-identifier>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<payment-amount>" & cpy!PD & "</payment-amount>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<bucket-transaction-type>" & cpy![PayType] & "</bucket-transaction-type>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<tendered-date>" & Format(cpy!ED, "yyyy-mm-dd") & "</tendered-date>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<entered-date>" & Format(cpy!ED, "yyyy-mm-dd") & "</entered-date>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<payment-location-code>" & cpy!PL & "</payment-location-code>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<payment-memo-code>" & cpy!MC & "</payment-memo-code>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<payment-method>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(12) & "<payment-method-type-code>" & cpy!PI & "</payment-method-type-code>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "</payment-method>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<payment-comments>" & [Forms]![adj_batch_01_main]![JIRA] & " Payment Batch</payment-comments>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "</consumer-payment>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "<consumer-identifier>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(10) & "<agency-identifier>" & cpy!PayConsumerNumber & "</agency-identifier>"
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & "</consumer-identifier>"
                    
                        '   Add account details
                            Set apy = dB.OpenRecordset("SELECT DISTINCT PayConsumerNumber, PayAccountNumber, SUM(PayAmt) AS capj_amt, PayType FROM payment_info WHERE xProc = 'N' AND xUser = '" & u!toolInitials & "' AND PayConsumerNumber = '" & cpy!PayConsumerNumber & "' AND xType = 'ADJ' GROUP BY PayConsumerNumber, PayAccountNumber, PayType ORDER BY PayAccountNumber;")
                            
                                axml = "<consumer-account-payment-imports>"
                            
                                Do Until apy.EOF
                            
                                Set fin = dB.OpenRecordset("SELECT * FROM payment_info WHERE xProc = 'N' AND xUser = '" & u!toolInitials & "' AND PayConsumerNumber = '" & apy!PayConsumerNumber & "' AND PayAccountNumber = '" & apy!PayAccountNumber & "' AND xType = 'ADJ' ORDER BY PayBucketName;")
                                
                                axml = axml & Chr(13) & Chr(10) & Space(10) & "<consumer-account-payment-import>"
                                axml = axml & Chr(13) & Chr(10) & Space(12) & "<consumer-account-payment>"
                                axml = axml & Chr(13) & Chr(10) & Space(14) & "<payment-amount>" & apy!capj_amt & "</payment-amount>"
                                axml = axml & Chr(13) & Chr(10) & Space(14) & "<bucket-transaction-type>" & apy!PayType & "</bucket-transaction-type>"
                                axml = axml & Chr(13) & Chr(10) & Space(12) & "</consumer-account-payment>"
                                    axml = axml & Chr(13) & Chr(10) & Space(12) & "<consumer-account-financial-details>"
                                      Do Until fin.EOF
                                        axml = axml & IIf(IsNull(fin!PayBucketName), "", Chr(13) & Chr(10) & Space(14) & "<consumer-account-financial-detail>")
                                        axml = axml & IIf(IsNull(fin!PayBucketName), "", Chr(13) & Chr(10) & Space(16) & "<bucket-name>" & fin!PayBucketName & "</bucket-name>")
                                        axml = axml & IIf(IsNull(fin!PayBucketName), "", Chr(13) & Chr(10) & Space(16) & "<bucket-transaction-type>" & fin!PayType & "</bucket-transaction-type>")
                                        axml = axml & IIf(IsNull(fin!PayBucketName), "", Chr(13) & Chr(10) & Space(16) & "<transaction-amount>" & fin!PayBucketAmt & "</transaction-amount>")
                                        axml = axml & IIf(IsNull(fin!PayBucketName), "", Chr(13) & Chr(10) & Space(14) & "</consumer-account-financial-detail>")
                                      fin.MoveNext
                                      Loop
                                    axml = axml & Chr(13) & Chr(10) & Space(12) & "</consumer-account-financial-details>"
                                axml = axml & Chr(13) & Chr(10) & Space(12) & "<consumer-account-identifier>"
                                axml = axml & Chr(13) & Chr(10) & Space(14) & "<agency-identifier>" & apy!PayAccountNumber & "</agency-identifier>"
                                axml = axml & Chr(13) & Chr(10) & Space(12) & "</consumer-account-identifier>"
                                axml = axml & Chr(13) & Chr(10) & Space(10) & "</consumer-account-payment-import>"
                                
                                apy.MoveNext
                                Loop
                                
                                axml = axml & Chr(13) & Chr(10) & Space(8) & "</consumer-account-payment-imports>"
                                
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(8) & axml
                        
                    xmlb = xmlb & Chr(13) & Chr(10) & Space(6) & "</consumer-payment-import>"
                    
                    f.Write xmlb
                    xmlb = ""
                    axml = ""
                    seq = seq + 1
                    
                    cpy.MoveNext
                    Loop
                      
        '   End of Adjs
            
                f.Write Chr(13) & Chr(10) & "       </consumer-payment-imports>"
                f.Write Chr(13) & Chr(10) & "</consumer-payment-import-job>"
                
                dB.Execute "UPDATE payment_info SET xProc = 'Y' WHERE xUser = '" & u!toolInitials & "';"
                
                f.Close
        
        
        '  Register the transaction file
            
                jBody = "{"
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileName"":""" & pay_batch & ".txt"","
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileType"":""PAYMENTIMPORT"""
                jBody = jBody & Chr(13) & Chr(10) & "}"
            
                    httpReq.Open "POST", envURL & "/fileregistry", False, api!xtsUser, api!xtsPass
                    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                    httpReq.setRequestHeader "Authorization", api!xtsAuth
                    httpReq.sEnd (jBody)
                    
                    response = httpReq.responseText
                    reg_id = Mid(response, 46, 6)
                    
                    'Debug.Print response
                          
            '  Batch & import the transaction file
                jBody = "{"
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileRegistryId"":" & reg_id & ","
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """batchName"":""" & pay_batch & ""","
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """processPostPaymentBatch"":Y"
                jBody = jBody & Chr(13) & Chr(10) & "}"
            
                    httpReq.Open "POST", envURL & "/consumerfinancialimportbatches", False, api!xtsUser, api!xtsPass
                    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                    httpReq.setRequestHeader "Authorization", api!xtsAuth
                    httpReq.sEnd (jBody)
                      
               '  Debug.Print jBody
                
               response = httpReq.responseText
               'Debug.Print response
               
               cpb_id = Mid(response, 39, 6)
               jBody = ""
                   
        '   Wait for the transaction batch status to update to ACTIVE
            Do Until awt >= 10000
            strSQL = "SELECT * FROM vw_xfacts_btch_stts WHERE btch_id = " & cpb_id & ";"
            
                Select Case PayEnv
                
                    Case "test"
                        OpenDMrect rs, strSQL
                        
                    Case "stage"
                        OpenDMrecg rs, strSQL
                        
                    Case "prod"
                        OpenDMrecs rs, strSQL
                        
                End Select
                
                awt = awt + IIf(rs!btch_stts = "ACTIVE", 10000, 1)
                awt = awt + IIf(rs!btch_stts = "FAILED", 10000, 1)
                
                del = Timer + 10
                Do While Timer < del
                    DoEvents
                Loop
                
            Loop
            awt = 0
            
        '   Post the transaction batch when the status switches to ACTIVE
            Select Case rs!btch_stts
            
                Case "ACTIVE"
            
                    httpReq.Open "POST", envURL & "/consumerfinancialimportbatches/" & cpb_id & "/post", False, api!xtsUser, api!xtsPass
                    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                    httpReq.setRequestHeader "Authorization", api!xtsAuth
                    httpReq.sEnd (jBody)
                    
                    pst = pst + 1
                    
                Case "FAILED"
                
                    MsgBox "The financial transaction batch file has failed to upload to Debt Manager. Please check the Financial Transactions tab to troubleshoot the issue.", vbInformation, "BATCH UPLOAD FAILED"
                    pst = 0
                    
                Case Else
                
                    MsgBox "This process has timed out while attempting to upload the transaction batch file to Debt Manager. Please check the Financial Transactions tab to troubleshoot the issue.", vbInformation, "BATCH UPLOAD TIMEOUT"
                    pst = 0
                    
             End Select
             
             rs.Close
             
        '   Wait for the transaction batch status to update to POSTED
            If pst > 0 Then
            
                Do Until awt >= 10000
                strSQL = "SELECT * FROM vw_xfacts_btch_stts WHERE btch_id = " & cpb_id & ";"
                
                Select Case PayEnv
                
                    Case "test"
                        OpenDMrect rs, strSQL
                        
                    Case "stage"
                        OpenDMrecg rs, strSQL
                        
                    Case "prod"
                        OpenDMrecs rs, strSQL
                        
                End Select
                    
                    awt = awt + IIf(rs!btch_stts = "POSTED", 10000, 1)
                    awt = awt + IIf(rs!btch_stts = "PARTIAL", 10000, 1)
                    
                    del = Timer + 10
                    Do While Timer < del
                        DoEvents
                    Loop
                    
                Loop
                
            End If
        
            If rs!btch_stts = "POSTED" Then
            
                MsgBox "Payment batch " & pay_batch & " has been posted in Debt Manager. Please check the Financial Transactions tab for this batch to see if any further steps are needed.", vbInformation, "BATCH POSTED"
                dB.Execute "UPDATE payment_info SET xProc = 'Y', xComment = '" & Now() & " - Payment submitted for posting' WHERE xProc = 'N' AND xType = 'CHK' AND xUser = '" & u!toolInitials & "';"
                
                Else
                
                MsgBox "The financial transaction batch has failed to post in Debt Manager. Please check the Financial Transactions tab to troubleshoot the issue.", vbInformation, "BATCH POSTING FAILED"
                    
            End If
            
        rs.Close
    
    dB.Close

End Function

                    awt = awt + IIf(fst!stts_cd = 5, 10000, 1)
                    
                    del = Timer + 10
                    Do While Timer < del
                        DoEvents
                    Loop
                    
                Loop
                
            End If
        
            If fst!stts_cd <> 4 Then
            
                MsgBox "The financial transaction batch has failed to post in Debt Manager. Please check the Financial Transactions tab to troubleshoot the issue.", vbInformation, "BATCH POSTING FAILED"
                
            Else
'--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
        ' POST AR EVENTS
'--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

               'Set AR BDL location
                fileLoc = Left(fileLoc, Len(fileLoc) - 9) & "\bdl\"
           '    Create the empty file for writing to in the location specified above
                BDL_arfile = [Forms]![pay_bal_rplc_main]![JIRA] & "_ar_event_bdl_" & Format(Now(), "yyyyMMdd") & CLng(Timer) & ".txt"
                DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", fileLoc & BDL_arfile
                
                '   Get the count of records to be written from the table
                Set LineItem = dB.OpenRecordset("SELECT * FROM payment_info WHERE xProc = 'N' AND xType = 'RPL' AND xUser = '" & u!toolInitials & "';")
                
                    LineItem.MoveLast
                    cnt = LineItem.RecordCount
                    LineItem.MoveFirst
        
           '    Compose the XML file header
                hdr = "<?xml version=""1.0"" encoding=""UTF-8""?>"
                hdr = hdr & Chr(13) & Chr(10) & "<dm_data xmlns=""http://www.fico.com/xml/debtmanager/data/v1_0"">"
                hdr = hdr & Chr(13) & Chr(10) & "       <header>"
                hdr = hdr & Chr(13) & Chr(10) & "           <sender_id_txt>Organization</sender_id_txt>"
                hdr = hdr & Chr(13) & Chr(10) & "           <target_id_txt>" & PayEnv & "</target_id_txt>"
                hdr = hdr & Chr(13) & Chr(10) & "           <batch_id_txt>" & [Forms]![pay_bal_rplc_main]![JIRA] & "</batch_id_txt>"
                hdr = hdr & Chr(13) & Chr(10) & "       <operational_transaction_type>CONSUMER_ACCOUNT_AR_LOG</operational_transaction_type>"
                hdr = hdr & Chr(13) & Chr(10) & "           <total_count>" & cnt & "</total_count>"
                hdr = hdr & Chr(13) & Chr(10) & "           <creation_data>" & Format(Date, "yyyy-mm-dd") & "T" & Format(Time(), "Short time") & ":00</creation_data>"
                hdr = hdr & Chr(13) & Chr(10) & "       </header>"
                hdr = hdr & Chr(13) & Chr(10) & "       <operational_transaction_data>"
                hdr = hdr & Chr(13) & Chr(10) & "           <cnsmr_accnt_ar_log_operational_transaction_data>"
            
            '   Write the header to file
                Set fs = CreateObject("Scripting.FileSystemObject")
                Set f = fs.OpenTextFile(fileLoc & BDL_arfile, ForAppending, TristateFalse)
                
                f.Write hdr
                
            '   Begin composing the AR Event XML records
                seq = 1
                
                    Do Until LineItem.EOF
                    
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(16) & "<cnsmr_accnt_ar_log seq_no=""" & seq & """ type=""CONSUMER_ACCOUNT_AR_LOG"">"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<cnsmr_idntfr_agncy_id>" & LineItem!PayConsumerNumber & "</cnsmr_idntfr_agncy_id>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<cnsmr_accnt_idntfr_agncy_id>" & LineItem!PayAccountNumber & "</cnsmr_accnt_idntfr_agncy_id>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<actn_cd_shrt_val_txt>CC</actn_cd_shrt_val_txt>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<rslt_cd_shrt_val_txt>CC</rslt_cd_shrt_val_txt>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<cnsmr_accnt_ar_mssg_txt>" & [Forms]![pay_bal_rplc_main]![JIRA] & ": Balance adjusted to zero</cnsmr_accnt_ar_mssg_txt>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(20) & "<cnsmr_accnt_ar_log_crt_usr_nm>apiuser</cnsmr_accnt_ar_log_crt_usr_nm>"
                        xmlb = xmlb & Chr(13) & Chr(10) & Space(16) & "</cnsmr_accnt_ar_log>"
                    
                    f.Write xmlb
                    xmlb = ""
                    seq = seq + 1
                    
                    LineItem.MoveNext
                    
                    Loop
             '  CLose the last XML Nodes and write to file
                f.Write Chr(13) & Chr(10) & "           </cnsmr_accnt_ar_log_operational_transaction_data>"
                f.Write Chr(13) & Chr(10) & "       </operational_transaction_data>"
                f.Write Chr(13) & Chr(10) & "</dm_data>"
                
                f.Close
                     
            '   Register the BDL file
            
                jBody = "{"
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileName"":""" & BDL_arfile & ""","
                jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileType"":""BDL_IMPORT"""
                jBody = jBody & Chr(13) & Chr(10) & "}"
            
                    httpReq.Open "POST", envURL & "/fileregistry", False, api!xtsUser, api!xtsPass
                    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                    httpReq.setRequestHeader "Authorization", api!xtsAuth
                    httpReq.sEnd (jBody)
            
                    response = httpReq.responseText
                    reg_id = Mid(response, 46, 6)
                    
                jBody = ""
            
                    'Debug.Print response
            
            '   Import the registered BDL file
                httpReq.Open "POST", envURL & "/fileregistry/" & reg_id & "/bdlimport", False, api!xtsUser, api!xtsPass
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", api!xtsAuth
                httpReq.sEnd (jBody)
                
                response = httpReq.responseText
                
                MsgBox "Payment batch " & pay_batch & " has been posted in Debt Manager. Please check the Financial Transactions tab for this batch to see if any further steps are needed.", vbInformation, "BATCH POSTED"
                dB.Execute "UPDATE payment_info SET xProc = 'Y', xComment = '" & Now() & " - Payment submitted for posting' WHERE xProc = 'N' AND xType = 'RPL' AND xUser = '" & u!toolInitials & "';"
        
            End If
        
        End If
    
    dB.Close
End Function
