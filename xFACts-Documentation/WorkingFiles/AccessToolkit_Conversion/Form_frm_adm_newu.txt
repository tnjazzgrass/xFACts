Option Compare Database

Private Sub usrAddNew_Click()
Dim err As Integer
Dim dB As Database
Dim stDocName As String
Dim stLinkCriteria As String
Dim uName As String, uRole As String
Dim lkName As Recordset, lkInit As Recordset, scrbld As Recordset

    Set dB = CurrentDb

'   Error Check - make sure form is filled out
    err = 0
    
    If IsNull(Me.nUser) = True Then
        MsgBox "Please enter a new username to proceed.", vbOKOnly, "REQUIRED FIELD MISSING"
        err = err + 1
    End If
    If IsNull(Me.nInit) = True Then
        MsgBox "Please enter a new set of initials to proceed.", vbOKOnly, "REQUIRED FIELD MISSING"
        err = err + 1
    End If

    'If everything is filled in, then check to make sure the user doesn't already exist
    If err = 0 Then
        
        Set lkName = dB.OpenRecordset("SELECT xtsUsername FROM xtsUsers WHERE xtsUsername = '" & Me.nUser & "';")
        
            If lkName.EOF Then
            
                'Also check to make sure the entered initials are unique
                Set lkInit = dB.OpenRecordset("SELECT xtsInitials FROM xtsUsers WHERE xtsInitials = '" & Me.nInit & "';")
                
                    If lkInit.EOF Then
                    
                    '   If it all looks good, then add the user and copy over their forms
                        DoCmd.Hourglass True
                        dB.Execute ("INSERT INTO xtsUsers (xtsUsername, xtsRole, xtsActive, xtsInitials, xtsDate) SELECT '" & Me.nUser & "', '" & Me.nRole & "', -1, '" & Me.nInit & "', NOW();")
                    '   Select the screens to build
                        Set scrbld = dB.OpenRecordset("SELECT bldScreen, bldFunction FROM xtsBuild WHERE bldFunction = 'All';")
    
                            Do Until scrbld.EOF
        
                                DoCmd.CopyObject , Left(scrbld!bldscreen, Len(scrbld!bldscreen) - 3) & Me.nInit, acForm, scrbld!bldscreen
    
                            scrbld.MoveNext
                            Loop

                        DoCmd.Hourglass False
                        
                    MsgBox "The new user has been added to the system.", vbOKOnly, "USER ADDED"
                        
                    End If
                
            Else
        
                MsgBox "Username and Initials must be unique in order to add a user.", vbOKOnly, "INVALID DATA"
                
            End If
            
    End If
    
    dB.Close
    
End Sub

Private Sub usrNewBack_Click()

    DoCmd.OpenForm "frm_adm_vwu", acNormal
    DoCmd.Close acForm, "frm_adm_newu", acSaveNo

End Sub
