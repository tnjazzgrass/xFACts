Private Sub api_MailRet_Click()
Dim del As Long
Dim dB As Database
Dim rst1 As Recordset
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String
Dim sURL As String, sUser As String, sPass As String, sAuth As String

    Set dB = CurrentDb
    Set rst1 = dB.OpenRecordset("SELECT * FROM delete_mail_returns WHERE Processed = 'N' ORDER BY ConsAgencyID")

'TEST
    If [Forms]![Call REST API]![runEnv] = 1 Then
    
            sURL = "http://dm-test-app:8080/dm-rest-services/api/"
            
            Do Until rst1.EOF
    
                httpReq.Open "DELETE", sURL & "consumers/" & rst1!ConsAgencyID & "/addressmailreturn", False, [Forms]![Call REST API]![APIuser], [Forms]![Call REST API]![APIpass]
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", [Forms]![Call REST API]![authCode]
                httpReq.sEnd
    
                response = httpReq.responseText
                sql = "UPDATE delete_mail_returns SET API_Response = '" & response & "', Processed = 'Y' WHERE ConsAgencyID = " & rst1!ConsAgencyID & ";"
                dB.Execute (sql)
            
            rst1.MoveNext
            Loop
            
        dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('delMailRet', 'TEST', '" & Environ("username") & "', Now())"
        
        MsgBox "The API calls to delete mail returns in TEST have been submitted", vbInformation, "APIs IN TEST SUBMITTED"
        
    End If

'STAGE
    If [Forms]![Call REST API]![runEnv] = 2 Then
    
            sURL = "http://dm-stage-app:8080/dm-rest-services/api/"
            
            Do Until rst1.EOF
    
                httpReq.Open "DELETE", sURL & "consumers/" & rst1!ConsAgencyID & "/addressmailreturn", False, [Forms]![Call REST API]![APIuser], [Forms]![Call REST API]![APIpass]
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", [Forms]![Call REST API]![authCode]
                httpReq.sEnd
    
                response = httpReq.responseText
                sql = "UPDATE delete_mail_returns SET API_Response = '" & response & "', Processed = 'Y' WHERE ConsAgencyID = " & rst1!ConsAgencyID & ";"
                dB.Execute (sql)
            
            rst1.MoveNext
            Loop
            
        dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('delMailRet', 'STAGE', '" & Environ("username") & "', Now())"
        
        MsgBox "The API calls to delete mail returns in STAGE have been submitted", vbInformation, "APIs IN STAGE SUBMITTED"
        
    End If
    
'PROD
    If [Forms]![Call REST API]![runEnv] = 3 Then
    
            sURL = "http://dm-prod-app:8080/dm-rest-services/api/"
            
            Do Until rst1.EOF

                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"

                dB.Execute (sql)
            
            rst1.MoveNext

        MsgBox "The API calls to delete mail returns in PRODUCTION have been submitted", vbInformation, "APIs IN PROD SUBMITTED"
