Option Compare Database
Private Sub addAcctPay_Click()
Dim dB As Database
Dim rec As Recordset, rs As ADODB.Recordset
Dim comm As String, sql As String, err As Integer

    Set dB = CurrentDb

'ERROR CHECK
    err = 0
    
'ERROR  Consumer number not filled in
        If IsNull(Me.adj_cnsmr) = True Then
            MsgBox "Please enter a consumer number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
            err = err + 1
        End If
        
'ERROR  Account number not filled in
        If err = 0 And IsNull(Me.adj_accnt) = True Then
            MsgBox "Please enter an account number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
            err = err + 1
        End If
        
'ERROR  Payment amount not filled in
        If err = 0 And IsNull(Me.adj_amt) = True Then
            MsgBox "Please enter a valid payment amount to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
            err = err + 1
        End If
        
'Parse comment
        If IsNull(Me.adj_cmnt) Then
            comm = [Forms]![adj_batch_01_main]![JIRA] & " Adjustment"
        Else
            comm = Me.adj_cmnt
        End If
                 
'NO ERRORS  Pull accounts into temp table to parse adjustments
            If err = 0 Then
        
                sql = "INSERT INTO payment_info (PayConsumerNumber, PayAccountNumber, PayAmt, PayBucketName, PayBucketAmt, PayType, PayLocation, PayEnterDate, PayTenderDate, PayMemoCode, PayComment, xUser, xType)"
                sql = sql & " VALUES ( '" & Me.adj_cnsmr & "', " & Me.adj_accnt & ", " & Me.adj_amt & ", '" & Me.adj_bckt & "', '" & Me.adj_amt & "', 'ADJUSTMENT', '" & Me.adj_location & "', '" & Format(Me.adj_enter, "yyyy-MM-dd") & "', '" & Format(Me.adj_enter, "yyyy-MM-dd") & "', 'NA', '" & comm & "', '" & [Forms]![adj_batch_01_main]![usr] & "', 'ADJ' )"
                dB.Execute sql
    
            End If
        
    dB.Close
    
    Forms![adj_batch_01_main]![adj_batch_02_detail].Form.Requery
    DoCmd.Close acForm, "adj_batch_03_add_adj", acSaveNo

End Sub
