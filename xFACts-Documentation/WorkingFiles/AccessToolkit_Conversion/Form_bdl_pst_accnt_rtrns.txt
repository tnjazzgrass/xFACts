Option Compare Database
Private Sub backRtrnAccnt_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_accnt_rtrns", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.bdl_accnt_rtrn_dtl.Requery
    
End Sub

Private Sub pstRtrnAccts_Click()
Dim dB As Database
Dim tgs As Recordset

    Set dB = CurrentDb
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM bdl_accnt_rtrn WHERE bdl_usr = '" & Me.usr.Value & "' AND bdl_proc = 'N';")
    
        If tgs!recs = 0 Then
        
            MsgBox "Please enter consumer and account IDs to return before attempting to run the BDL.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
            
        Else

            If Me.JIRA = "IT-" Or Left(Me.JIRA, 3) <> "IT-" Then
                
                    MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                    
            Else
            
                Select Case Me.runEnv
            
                    Case 1
                    
                        ReturnAccounts ("test")
                
                    Case 2
                    
                        ReturnAccounts ("stage")
                
                    Case 3
                    
                        ReturnAccounts ("prod")
                        
                End Select
            
            End If
            
        End If
        
    dB.Close
    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_accnt_rtrns", acSaveNo

End Sub
