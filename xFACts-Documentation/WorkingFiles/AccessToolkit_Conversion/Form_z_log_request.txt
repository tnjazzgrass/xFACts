VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "Form_z_log_request"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = True
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Public Loops As Integer

Private Sub Form_Load()
 Me.TimerInterval = 1000
 Form_Timer
End Sub

Private Sub Form_Timer()

Static StartTime As Date

Dim SecondsToCount As Integer

SecondsToCount = 180 'Set this variable to the total number of seconds to count down

If Loops = 0 Then StartTime = Time
 
 min = (SecondsToCount - DateDiff("s", StartTime, Time)) \ 60
 Sec = (SecondsToCount - DateDiff("s", StartTime, Time)) Mod 60
 Me.txtMinsSecs.Value = min & ":" & Format(Sec, "00")
 Loops = Loops + 1

 If Me.txtMinsSecs.Value = "0:00" Then
   DoCmd.Quit
 End If

End Sub

Private Sub hideNote_Click()

    Me.Visible = False
    
    MsgBox "The Apps Toolkit is scheduled close down shortly, please save your work and exit as soon as possible.", vbInformation, "Maintenance Shutdown"

End Sub
