Private Sub impNB_Click()
Dim dB As Database
Dim tgs As Recordset

    Set dB = CurrentDb
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM crdtr_fee_dm_accnt WHERE nb_proc = 'N';")
    
        If tgs!recs = 0 Then
        
            MsgBox "Please create new accounts before attempting to run the import.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
            
        Else
            
            Select Case Me.runEnv
        
                Case 1
                
                    NewCredAcct ("test")
            
                Case 2
                
                    NewCredAcct ("stage")
            
                Case 3
                
                    NewCredAcct ("prod")
                    
            End Select
            
            DoCmd.OpenForm "fxd_fee_02_config_02", acNormal
            DoCmd.Close acForm, "fxd_fee_02_config_03", acSaveNo
            
        End If
        
    dB.Close

End Sub
