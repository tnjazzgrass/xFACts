Option Compare Database
Private Sub backCnsmrUpdt_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_phn_updts", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.bdl_pst_phn_updts_dtl.Requery
    
End Sub
Private Sub pstPhnUpdt_Click()
Dim dB As Database
Dim tgs As Recordset

    Set dB = CurrentDb
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM bdl_phn WHERE bdl_usr = '" & Me.usr.Value & "' AND bdl_proc = 'N';")
    
        If tgs!recs = 0 Then
        
            MsgBox "Please enter new phone records before attempting to run the BDL.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
            
        Else

            If Me.JIRA = "SD-" Or Left(Me.JIRA, 3) <> "SD-" Then
                
                    MsgBox "Please enter a JIRA Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                    
            Else
            
                Select Case Me.runEnv
            
                    Case 1
                    
                        PostPhnUpdts ("test")
                
                    Case 2
                    
                        PostPhnUpdts ("stage")
                
                    Case 3
                    
                        PostPhnUpdts ("prod")
                        
                End Select
            
            End If
            
        End If
        
    dB.Close

End Sub
