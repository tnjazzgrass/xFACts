Private Sub vopClsPst_Click()

Private Sub vopPostAR_Click()

        Select Case Me.runEnv
        
            Case 1
            
                DoCmd.OpenQuery "voapps_form_ars", acViewNormal, acAdd
                PostVOAppARs ("test")
            
            Case 2
            
                DoCmd.OpenQuery "voapps_form_ars", acViewNormal, acAdd
                PostVOAppARs ("stage")
            
            Case 3

                DoCmd.OpenForm "voapps_pst_prod", acNormal
                DoCmd.Close acForm, "voapps_04postARs", acSaveNo
            
        End Select

End Sub
