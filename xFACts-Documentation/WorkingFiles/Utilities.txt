Public Enum rrCursorType

    rrOpenForwardOnly = adOpenForwardOnly

End Enum

Public Enum rrLockType
    rrLockOptimistic = adLockOptimistic
    rrLockReadOnly = adLockReadOnly
End Enum
Function OpenDMrecg(rs As ADODB.Recordset, strSQL As String, Optional rrCursor As rrCursorType, Optional rrLock As rrLockType, Optional bolClientSide As Boolean) As ADODB.Recordset
    
    If cong.State = adStateClosed Then
        cong.ConnectionString = "Provider=SQLOLEDB.1;Server=avg-stage-lsnr;Database=Applications;TRUSTED_CONNECTION=YES;"
        cong.Open
    End If

    Set rs = New ADODB.Recordset
    With rs
        .ActiveConnection = cong
        If bolClientSide Then
            .CursorLocation = adUseClient
        Else
            .CursorLocation = adUseServer
        End If
        .CursorType = IIf((rrCursor = 0), adOpenStatic, rrCursor)
        .LockType = IIf((rrLock = 0), adLockReadOnly, rrLock)
        .Open strSQL
        If .EOF And .BOF Then
            NoRecords = True
            Exit Function
        End If
    End With
    
End Function

    With comm
        .ActiveConnection = con
        .CommandText = strSQL
        .Execute lngRecordsAffected
        'Debug.Print lngRecordsAffected
    End With
    ExecDMcmd = True
    
ExitProcedure:
    'Set Comm = Nothing

ErrorHandler:

Function MergeConsumers(frmRec As String)
Dim del As Long
Dim dB As Database
Dim rst1 As Recordset
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String

    Set dB = CurrentDb
    Set rst1 = dB.OpenRecordset("SELECT * FROM merge_consumers_string WHERE ResponseString IS NULL ORDER BY ConsumerName, TargetAgencyID, SourceAgencyID")
    
    Do Until rst1.EOF
        
        jBody = "{"
        jBody = jBody & Chr(13) & Chr(10) & """consumerAgencyId"": " & rst1!TargetAgencyID & ","
        jBody = jBody & Chr(13) & Chr(10) & """linkedAccounts"": " & rst1!AccountString
        jBody = jBody & Chr(13) & Chr(10) & "}"
    
            httpReq.Open "POST", Me.APIurl & "consumers/" & rst1!SourceAgencyID & "/merge", False, Me.APIuser, Me.APIpass
            httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
            httpReq.setRequestHeader "Authorization", Me.authCode
            httpReq.sEnd (jBody)
        
        response = httpReq.responseText
        sql = "UPDATE merge_consumers_string SET ResponseString = '" & response & "' WHERE SourceAgencyID = " & rst1!SourceAgencyID
        dB.Execute (sql)
       
       del = Timer + 1
       Do While Timer < del
           DoEvents
       Loop
    
    rst1.MoveNext
    Loop
    
    MsgBox response, vbInformation, "WEBSERVICE RESPONSE"

End Function

Function PostVOAppARs(VOenv As String)
Dim fs, f
Dim httpReq As New XMLHTTP60
Dim fld As Field, nfld As Field
Dim cnt As Integer, seq As Integer
Dim LineItem As Recordset, c As Recordset
Dim BDL_arfile As String, hdr As String, xmlb As String, reg_id As String
Dim fileLoc As String, envURL As String, response As String, jBody As String

    Set dB = CurrentDb
    
    '   Set environment specific variables
        Select Case VOenv
        
            Case "test"
            
                fileLoc = "\\dm-test-app\e$\dmfs\import\bdl\"
                envURL = "http://dm-test-app:8080/dm-rest-services/api"
                
            Case "stage"
            
                fileLoc = "\\dm-stage-app3\dmfs\import\bdl\"
                envURL = "http://dm-stage-app:8080/dm-rest-services/api"
                
            Case "prod"
            
                fileLoc = "\\dm-prod-app3\dmfs\import\bdl\"
                envURL = "http://dm-prod-app:8080/dm-rest-services/api"
                
        End Select
    
   '    Create the empty file for writing to in the location specified above
        BDL_arfile = "VOApps_File_" & Format(Now(), "yyyyMMdd") & CLng(Timer) & ".txt"
        DoCmd.TransferText acExportFixed, "BlankFileExp", "BlankFile", fileLoc & BDL_arfile

   '    Select the unprocessed records from the VOApps AR table
        Set LineItem = dB.OpenRecordset("SELECT * FROM voapps_ar_log WHERE var_proc = 'N';")
        
        '   Get the count of records to be written from the table
            LineItem.MoveLast
            cnt = LineItem.RecordCount
            LineItem.MoveFirst

   '    Compose the XML file header
        hdr = "<?xml version=""1.0"" encoding=""UTF-8""?>"
        hdr = hdr & Chr(13) & Chr(10) & "<dm_data xmlns=""http://www.fico.com/xml/debtmanager/data/v1_0"">"
        hdr = hdr & Chr(13) & Chr(10) & "       <header>"
        hdr = hdr & Chr(13) & Chr(10) & "           <sender_id_txt>VOApps</sender_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "           <target_id_txt>FAC Debt Manager</target_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "           <batch_id_txt>" & BDL_arfile & "</batch_id_txt>"
        hdr = hdr & Chr(13) & Chr(10) & "       <operational_transaction_type>CONSUMER_ACCOUNT_AR_LOG</operational_transaction_type>"
        hdr = hdr & Chr(13) & Chr(10) & "           <total_count>" & cnt & "</total_count>"
        hdr = hdr & Chr(13) & Chr(10) & "           <creation_data>" & Format(Date, "yyyy-mm-dd") & "T" & Format(Time(), "Short time") & ":00</creation_data>"
        hdr = hdr & Chr(13) & Chr(10) & "       </header>"
        hdr = hdr & Chr(13) & Chr(10) & "       <operational_transaction_data>"
        hdr = hdr & Chr(13) & Chr(10) & "           <cnsmr_accnt_ar_log_operational_transaction_data>"
    
    '   Write the header to file
        Set fs = CreateObject("Scripting.FileSystemObject")
        Set f = fs.OpenTextFile(fileLoc & BDL_arfile, ForAppending, TristateFalse)
        
        f.Write hdr
        
    '   Begin composing the AR Event XML records
        seq = 1
        
            Do Until LineItem.EOF
            
                xmlb = xmlb & Chr(13) & Chr(10) & Space(16) & "<cnsmr_accnt_ar_log seq_no=""" & seq & """ type=""CONSUMER_ACCOUNT_AR_LOG"">"
                xmlb = xmlb & IIf(Len(LineItem!cnsmr_idntfr_agncy_id > 0), Chr(13) & Chr(10) & Space(20) & "<cnsmr_idntfr_agncy_id>" & LineItem!cnsmr_idntfr_agncy_id & "</cnsmr_idntfr_agncy_id>", "")
                xmlb = xmlb & IIf(Len(LineItem!actn_cd_shrt_val_txt > 0), Chr(13) & Chr(10) & Space(20) & "<actn_cd_shrt_val_txt>" & LineItem!actn_cd_shrt_val_txt & "</actn_cd_shrt_val_txt>", "")
                xmlb = xmlb & IIf(Len(LineItem!rslt_cd_shrt_val_txt > 0), Chr(13) & Chr(10) & Space(20) & "<rslt_cd_shrt_val_txt>" & LineItem!rslt_cd_shrt_val_txt & "</rslt_cd_shrt_val_txt>", "")
                xmlb = xmlb & IIf(Len(LineItem!cnsmr_accnt_ar_mssg_txt > 0), Chr(13) & Chr(10) & Space(20) & "<cnsmr_accnt_ar_mssg_txt>" & LineItem!cnsmr_accnt_ar_mssg_txt & "</cnsmr_accnt_ar_mssg_txt>", "")
                xmlb = xmlb & IIf(Len(LineItem!cnsmr_accnt_ar_log_crt_usr_nm > 0), Chr(13) & Chr(10) & Space(20) & "<cnsmr_accnt_ar_log_crt_usr_nm>" & LineItem!cnsmr_accnt_ar_log_crt_usr_nm & "</cnsmr_accnt_ar_log_crt_usr_nm>", "")
                xmlb = xmlb & Chr(13) & Chr(10) & Space(16) & "</cnsmr_accnt_ar_log>"
            
            f.Write xmlb
            xmlb = ""
            seq = seq + 1
            
            LineItem.MoveNext
            
            Loop
     '  CLose the last XML Nodes and write to file
        f.Write Chr(13) & Chr(10) & "           </cnsmr_accnt_ar_log_operational_transaction_data>"
        f.Write Chr(13) & Chr(10) & "       </operational_transaction_data>"
        f.Write Chr(13) & Chr(10) & "</dm_data>"
        
        f.Close
             
'   Register the BDL file

    Set c = dB.OpenRecordset("SELECT * FROM APIDeets;")

    jBody = "{"
    jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileName"":""" & BDL_arfile & ""","
    jBody = jBody & Chr(13) & Chr(10) & Space(2) & """fileType"":""BDL_IMPORT"""
    jBody = jBody & Chr(13) & Chr(10) & "}"

        httpReq.Open "POST", envURL & "/fileregistry", False, c!uName, c!pwd
        httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
        httpReq.setRequestHeader "Authorization", c!auth
        httpReq.sEnd (jBody)

        response = httpReq.responseText
        reg_id = Mid(response, 46, 6)
        
    jBody = ""

        'Debug.Print response

'   Import the registered BDL file
    httpReq.Open "POST", envURL & "/fileregistry/" & reg_id & "/bdlimport", False, c!uName, c!pwd
    httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
    httpReq.setRequestHeader "Authorization", c!auth
    httpReq.sEnd (jBody)
    
    response = httpReq.responseText
        
'   Mark all the records as processed
    dB.Execute ("UPDATE voapps_stg SET vstg_proc = 'Y' WHERE vstg_proc = 'N'")
    dB.Execute ("UPDATE voapps_ar_log SET var_proc = 'Y', var_post = NOW(), var_env = '" & VOenv & "' WHERE var_proc = 'N'")
    dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('postVOApp', '" & VOenv & "', '" & Environ("username") & "', Now())"
    dB.Close
        
    MsgBox "BDL file " & BDL_arfile & " has been submitted in Debt Manager. Check the Consumer/Account Data Processing Screen in DM to verify that it processed completely.", vbInformation, "AR FILE SUBMITTED"

End Function

Public Function RandBetween(Optional Lowest As Long = 1, Optional Highest As Long = 9999) As Integer
' Generates a random whole number within a given range

End Function

                        pst = 0
                        
                    Case Else
                    
                        MsgBox "This process has timed out while attempting to post the Reversal Batch in Debt Manager. Please check the Financial Transactions tab to troubleshoot the issue.", vbInformation, "BATCH POST TIMEOUT"
                        pst = 0
                        
                 End Select
                    
                    MsgBox "Payment batch " & pay_batch & " has been submitted in Debt Manager. Please check the Financial Transactions tab for this batch to see if any further steps are needed.", vbInformation, "BATCH SUBMITTED"
                    
                    dB.Execute "UPDATE payment_info SET xProc = 'Y' WHERE xProc = 'N' AND xUser = '" & u!toolInitials & "' AND xComment IS NULL;"
                    
        '    End If
                          
        '   END OF PAYMENT REALLOCATION
        '   ================================================================================================================================================================================
        
    End If
    
    dB.Close

End Function
Function fix100(cid As Long, envr As String)
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim rst1 As Recordset, rst2 As Recordset, rst3 As Recordset, rst4 As Recordset
Dim stLinkCriteria As String, sURL As String, response As String, jBody As String
Dim sql1 As String, sql2 As String, sql3 As String, aList As String, env As String
Dim uName As String, uRole As String, strSQL As String, sql As String, u As Recordset
Dim stDocName As String, del As Long, bizID As Long, aStat As String, apCnt As Long, gpCnt As Long, cgCnt As Long
Dim look As Recordset, rs As ADODB.Recordset, rsc As ADODB.Recordset, rsa As ADODB.Recordset, rsr As ADODB.Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT * FROM toolUsers WHERE toolUsername = '" & Environ("username") & "';")
    
    strSQL = "SELECT cn.consumer_id, CONSUMER_AGENCY_ID, account_id, ACCOUNT_AGENCY_ID, new_biz_batch_id, account_status_tag FROM vw_accnt_stts_smmry ac" & Chr(13) & Chr(10)
    strSQL = strSQL & "    INNER JOIN vw_cnsmr_stts_smmry cn ON cn.consumer_id = ac.consumer_id" & Chr(13) & Chr(10)
    strSQL = strSQL & "WHERE CONSUMER_AGENCY_ID = " & cid & ";"
    
    Select Case envr
        
        Case "test"
            OpenDMrect rs, strSQL
            sURL = "http://dm-test-app:8080/dm-rest-services/api/"
        
        Case "stage"
            OpenDMrecg rs, strSQL
            sURL = "http://dm-stage-app:8080/dm-rest-services/api/"
            
        Case "prod"
            OpenDMrecs rs, strSQL
            sURL = "http://dm-prod-app:8080/dm-rest-services/api/"
            
    End Select
    
        If rs.RecordCount = 0 Then

                MsgBox "This consumer number doesn't have enough accounts to be split using this tool.", vbOKOnly, "INELIGIBLE CONSUMER"
                rs.Close
                
        Else
        
            DoCmd.OpenForm "split100_wait", acNormal
            [Forms]![Call REST API].Visible = False
            
            del = Timer + 3
            Do While Timer < del
                DoEvents
            Loop
            
            dB.Execute ("DELETE FROM Split100")
            
            Do Until rs.EOF
            
                dB.Execute ("INSERT INTO Split100 ( ConsumerID, ConsumerNum, AccountID, AccountNum, NewBizBatch, CurrState ) VALUES (" & rs!consumer_id & ", " & rs!CONSUMER_AGENCY_ID & ", " & rs!account_id & ", " & rs!ACCOUNT_AGENCY_ID & ", " & rs!new_biz_batch_id & ", '" & rs!account_status_tag & "');")
            
            rs.MoveNext
            Loop
            
                dB.Execute ("UPDATE Split100 INNER JOIN dm_curr_stat ON Split100.CurrState = dm_curr_stat.Acct_Tag SET Split100.Status = [AccountStatus];")
            
            'Calculate order and grouping
               'First call up the consumer
                Set rst1 = dB.OpenRecordset("SELECT DISTINCT ConsumerNum AS consumer_number FROM Split100;")
                
                    Do Until rst1.EOF
                    
                       'Initialize counters
                        apCnt = 1
                        cgCnt = 1
                        gpCnt = 0
                        
                       'Select all accounts for current consumer
                        sql1 = "SELECT * FROM Split100 WHERE ConsumerNum = " & rst1!consumer_number & " ORDER BY Status, NewBizBatch, AccountID;"
                        Set rst2 = dB.OpenRecordset(sql1)
                        
                       'Keep track of account status
                        aStat = IIf(rst2.EOF, 0, rst2!Status)
                        
                           'Loop through accounts and number them for each consumer, break them into groups of about 100 while keeping accounts in the same new biz batches together
                            Do Until rst2.EOF
                            
                               'This numbers the groups of accounts per consumer (about 100 per group, keeping account status together)
                                If gpCnt < 91 Then  'aStat = rst2!Status Or
                                    cgCnt = cgCnt
                                    gpCnt = gpCnt + 1
                                Else
                                    cgCnt = cgCnt + 1
                                    gpCnt = 1
                                End If
                                
                                    sql2 = "UPDATE Split100 SET AccountPlace = " & apCnt & ", ConsumerGroup = " & cgCnt & ", GroupPlace = " & gpCnt & " WHERE AccountID = " & rst2!AccountID & ";"
                                    dB.Execute (sql2)
                                
                               'This just numbers the accounts for each consumer (1 to however many accounts the consumer originally had)
                                apCnt = apCnt + 1
                            
                           'Keep track of new biz batches
                            aStat = rst2!Status
                            
                            rst2.MoveNext
                            Loop
                            
                        rst2.Close
                    
                    rst1.MoveNext
                    Loop
                    
                    rst1.Close
                    
                   'The first group of accounts for each consumer doesn't need further processing
                    sql1 = "UPDATE Split100 SET Processed = 'X' WHERE ConsumerGroup = 1"
                    dB.Execute (sql1)
                   
            'Compose account string for API body
                 
                'Pull records that need API Calls
                 dB.Execute ("INSERT INTO Split_tmp_cons ( ConsID, ConsAgencyID, Grp ) SELECT DISTINCT ConsumerID, ConsumerNum, ConsumerGroup FROM Split100 GROUP BY ConsumerID, ConsumerNum, ConsumerGroup HAVING ConsumerGroup > 1;")
                    
                'Select accounts that will be split out for each API call
                 Set rst1 = dB.OpenRecordset("SELECT * FROM Split_tmp_Cons WHERE Processed = 'N' ORDER BY ConsID, Grp;")
                 
                    Do Until rst1.EOF
                    
                       'Loop through the records and compose the string of account IDs
                        sql2 = "SELECT * FROM Split100 WHERE ConsumerID = " & rst1!ConsID & " AND ConsumerGroup = " & rst1!Grp & " ORDER BY Status,NewBizBatch,GroupPlace;"
                        Set rst2 = dB.OpenRecordset(sql2)
                        
                            Do Until rst2.EOF
                            
                                aList = aList & "," & rst2!AccountNum
                            
                            rst2.MoveNext
                            Loop
                            
                       'Store the composed string for calling the API
                        sql3 = "UPDATE Split_tmp_cons SET AcctSplit = '" & Mid(aList, 2, Len(aList)) & "' WHERE ConsID = " & rst1!ConsID & " AND Grp = " & rst1!Grp & ";"
                        dB.Execute (sql3)
                            
                        aList = ""
                        sql2 = ""
                        sql3 = ""
                        
                    rst1.MoveNext
                    Loop
                        
                   'Loop through consumers that need to have accounts split and compose & make API calls
                    Set rst3 = dB.OpenRecordset("SELECT * FROM Split_tmp_Cons WHERE Processed = 'N' ORDER BY ConsID, Grp;")
                    
                       Do Until rst3.EOF
        
                           'Compose body of API call
                            jBody = "{"
                            jBody = jBody & Chr(13) & Chr(10) & """consumerAgencyId"":" & rst3!ConsAgencyID & ","
                            jBody = jBody & Chr(13) & Chr(10) & """accountIds"":[" & rst3!AcctSplit & "],"
                            jBody = jBody & Chr(13) & Chr(10) & """attachConsumers"":true,"
                            jBody = jBody & Chr(13) & Chr(10) & """attachLinkedConsumers"":false"
                            jBody = jBody & Chr(13) & Chr(10) & "}"
        
                           'Make API call
                            httpReq.Open "POST", sURL & "consumers/" & rst3!ConsAgencyID & "/split", False, [Forms]![Call REST API]![APIuser], [Forms]![Call REST API]![APIpass]
                            httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                            httpReq.setRequestHeader "Authorization", [Forms]![Call REST API]![authCode]
                            httpReq.sEnd (jBody)
        
                            response = response & Chr(13) & Chr(10) & httpReq.responseText
        
                           'Debug.Print response
        
                           'Update processed status
                            sql2 = "UPDATE Split_tmp_Cons SET Processed = 'Y', API_Response = '" & response & "' WHERE ConsID = " & rst3!ConsID & " AND Grp = " & rst3!Grp & ";"
                                dB.Execute (sql2)
        
                            sql3 = "UPDATE Split100 SET Processed = 'Y' WHERE ConsumerID = " & rst3!ConsID & " AND ConsumerGroup = " & rst3!Grp & ";"
                                dB.Execute (sql3)
        
                            del = Timer + 2
                            Do While Timer < del
                                DoEvents
                            Loop
        
                        rst3.MoveNext
                        Loop
                        
            DoCmd.Close acForm, "split100_wait", acSaveNo
            
            dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('Split100', '" & env & "', '" & Environ("username") & "', Now())"
    
            MsgBox "The split consumer/account APIs have been submitted" & Chr(13) & Chr(10) & response, vbInformation, "API CALLS COMPLETE"
            
        End If
                
    [Forms]![Call REST API].Visible = True
    
End Function
