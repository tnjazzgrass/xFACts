Private Sub addAdj_Click()
Dim err As Integer
    
    err = 0
    
    If Len(Me.JIRA) < 7 Then
        err = err + 1
    End If
    
    If Left(Me.JIRA, 3) <> "IT-" And Left(Me.JIRA, 3) <> "SD-" Then
        err = err + 1
    End If

    If err > 0 Then
                    
        MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                        
    Else
    
        DoCmd.OpenForm "adj_batch_03_add_adj", acNormal
        
    End If

End Sub
Private Sub backAdj_Click()

    DoCmd.OpenForm "fnncl_srvcs", acNormal
    DoCmd.Close acForm, "adj_batch_01_main", acSaveNo

End Sub

Private Sub delAdj_Click()
Dim dB As Database
Dim sql As String

    Set dB = CurrentDb
        sql = "DELETE * FROM payment_info WHERE PayID = " & Me.slct
    dB.Execute (sql)
    dB.Close
    
    Me.adj_batch_02_detail.Requery

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.adj_batch_02_detail.Requery
    
End Sub
Private Sub postAdjs_Click()
Dim dB As Database
Dim err As Integer
Dim tgs As Recordset

    
    err = 0
    
    If Len(Me.JIRA) < 7 Then
        err = err + 1
    End If
    
    If Left(Me.JIRA, 3) <> "IT-" And Left(Me.JIRA, 3) <> "SD-" Then
        err = err + 1
    End If

    If err > 0 Then
                    
        MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                        
    Else
    
        Set dB = CurrentDb
        Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM payment_info WHERE xProc = 'N' AND xType = 'ADJ' AND xUser = '" & Me.usr.Value & "' AND xComment IS NULL;")
        
            If tgs!recs = 0 Then
            
                MsgBox "Please enter adjustments before attempting to run the process.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
                
            Else
    
                Select Case Me.runEnv
            
                    Case 1
                    
                        PostAdjustments ("test")
                
                    Case 2
                    
                        PostAdjustments ("stage")
                
                    Case 3
                    
                        PostAdjustments ("prod")
                        
                End Select
                
            End If
           
        Me.adj_batch_02_detail.Requery
        Me.JIRA.Value = "SD-"
        dB.Close
        
    End If

End Sub
