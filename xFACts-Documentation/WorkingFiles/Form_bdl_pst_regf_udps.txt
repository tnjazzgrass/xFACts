Option Compare Database
Private Sub backRegFudps_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_regf_udps", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.bdl_regf_udp_dtl.Requery
    
End Sub
Private Sub pstRegFudps_Click()
Dim dB As Database
Dim tgs As Recordset, bal As Recordset

    Set dB = CurrentDb
    Set bal = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM bdl_regf_udp WHERE bdl_usr = '" & Me.usr.Value & "' AND bdl_proc = 'N' AND (serv_bal_due IS NULL OR cnsmr_accnt_idntfr_agncy_id IS NULL);")
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM bdl_regf_udp WHERE bdl_usr = '" & Me.usr.Value & "' AND bdl_proc = 'N';")
    
        If bal!recs > 0 Then
        
            MsgBox "Account Number and Balance at Time of Service are required fields, please make sure each record is complete.", vbCritical, "INVALID/MISSING DATA"
            
        Else
        
            If tgs!recs = 0 Then
            
                MsgBox "Please enter UDP details before attempting to run the BDL.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
                
            Else
    
                If Me.JIRA = "IT-" Or Left(Me.JIRA, 3) <> "IT-" Then
                    
                        MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                        
                Else
                
                    Select Case Me.runEnv
                
                        Case 1
                        
                            PostRegFUDPs ("test")
                    
                        Case 2
                        
                            PostRegFUDPs ("stage")
                    
                        Case 3
                        
                            PostRegFUDPs ("prod")
                            
                    End Select
                
                End If
                
            End If
            
        End If
        
    dB.Close
    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_regf_udps", acSaveNo

End Sub
