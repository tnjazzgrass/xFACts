Private Sub canSyncLow_Click()

    DoCmd.Close acForm, "runBalSync_low", acSaveNo
    [Forms]![Call REST API].Visible = True

End Sub

Private Sub runSyncLow_Click()
Dim del As Long
Dim dB As Database
Dim rst1 As Recordset
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String
Dim sURL As String, sUser As String, sPass As String, sAuth As String

    Set dB = CurrentDb
    
    sUser = [Forms]![Call REST API]![APIuser]
    sPass = [Forms]![Call REST API]![APIpass]
    sAuth = [Forms]![Call REST API]![authCode]
    
    Select Case [Forms]![Call REST API]![runEnv]

'TEST
    Case 1
    
        Set rst1 = dB.OpenRecordset("SELECT MAX(toolDate) AS lst FROM toolUse WHERE toolFunction = 'balSync' AND toolEnv = 'TEST';")
        
        If DateDiff("n", rst1!lst, Now()) < 20 Then
        
            MsgBox "This API was called less than 20 minutes ago, please wait until the last run has completed", vbInformation, "API RECENTLY CALLED"
            
        Else
    
            sURL = "http://dm-test-app:8080/dm-rest-services/api/scheduledjobs/UPDATE_BALANCES"

                httpReq.Open "POST", sURL, False, sUser, sPass
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", sAuth
                httpReq.sEnd

                response = httpReq.responseText
                
            dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('balSync', 'TEST', '" & Environ("username") & "', Now())"
            
            MsgBox response, vbInformation, "TEST WEBSERVICE RESPONSE"
            
        End If
        
'STAGE
    Case 2
    
        Set rst1 = dB.OpenRecordset("SELECT MAX(toolDate) AS lst FROM toolUse WHERE toolFunction = 'balSync' AND toolEnv = 'STAGE';")
        
        If DateDiff("n", rst1!lst, Now()) < 20 Then
        
            MsgBox "This API was called less than 20 minutes ago, please wait until the last run has completed", vbInformation, "API RECENTLY CALLED"
            
        Else
    
            sURL = "http://stage-app.fac.local:8080/dm-rest-services/api/scheduledjobs/UPDATE_BALANCES"

                httpReq.Open "POST", sURL, False, sUser, sPass
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", sAuth
                httpReq.sEnd

                response = httpReq.responseText
                
            dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('balSync', 'STAGE', '" & Environ("username") & "', Now())"
            
            MsgBox response, vbInformation, "STAGE WEBSERVICE RESPONSE"
            
        End If
        
    End Select
    
End Sub
