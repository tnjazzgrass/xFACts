Option Compare Database

Private Sub uBDLupdt_Click()
Dim dB As Database
Dim spCall As String

    spCall = "EXEC sp_DM_BDL_Force_Status_Update " & Me.idBDL
    
    ExecDMcmd spCall
    
    Set dB = CurrentDb
    
        dB.Execute "UPDATE dm_stat_bdl SET Fix = NULL;"
        
    dB.Close

    DoCmd.Close acForm, "frm_adm_bdl_fix", acSaveNo

End Sub

Private Sub xBDLupdt_Click()
Dim dB As Database

    Set dB = CurrentDb
    
        dB.Execute "UPDATE dm_stat_bdl SET Fix = NULL;"
        
    dB.Close

    DoCmd.Close acForm, "frm_adm_bdl_fix", acSaveNo

End Sub
