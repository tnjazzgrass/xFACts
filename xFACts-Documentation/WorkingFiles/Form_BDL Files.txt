Private Sub rollUDP_Click()
Dim sql As String
Dim dB As Database
Dim rst As Recordset

    Set dB = CurrentDb
    
    Set rst = dB.OpenRecordset("SELECT cnsmr_accnt_idntfr_agncy_id FROM bdl_accnt_tag;")
    
    Do Until rst.EOF
    
        sql = "UPDATE bdl_accnt_tag SET cnsmr_idntfr_agncy_id = '" & GetConsumerID(rst!cnsmr_accnt_idntfr_agncy_id) & "' WHERE cnsmr_accnt_idntfr_agncy_id = '" & rst!cnsmr_accnt_idntfr_agncy_id & "';"
        dB.Execute (sql)
        sql = ""
        
    rst.MoveNext
    Loop
    
    dB.Close
    
    MsgBox "All consumer IDs have been retrieved.", vbInformation, "DONE"
    
End Sub
