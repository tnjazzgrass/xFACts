Attribute VB_Name = "Form_split100_API"

Option Compare Database
Private Sub canSplit100_Click()

    DoCmd.Close acForm, "split100_API", acSaveNo
    [Forms]![Call REST API].Visible = True
    
End Sub

Private Sub split100a_Click()
Dim bizID As Long

    bizID = Me.c100
    
    Select Case [Forms]![Call REST API]![runEnv]
        
        Case 1
            DoCmd.Close acForm, "split100_API", acSaveNo
            Call fix100(bizID, "test")
            
        Case 2
            MsgBox "The STAGE Environment is not available.", vbOKOnly, "ENVIRONMENT INACTIVE"
            
        Case 3
            DoCmd.Close acForm, "split100_API", acSaveNo
            Call fix100(bizID, "prod")
            
    End Select
    
End Sub
