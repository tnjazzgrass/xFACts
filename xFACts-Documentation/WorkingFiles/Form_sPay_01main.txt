Private Sub closeSPay_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "sPay_01main", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim perm As Recordset
Dim uName As String, uRole As String

    uName = Environ("username")

    Set dB = CurrentDb
    
    Set perm = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & uName & "';")
    
        If perm.EOF Then
        
            MsgBox "This application has not been set up for user " & uName & ", please contact the Apps team for access.", vbOKOnly, "ACCESS DENIED"
            DoCmd.Quit acQuitSaveNone
            
        Else
        
            uRole = perm!xtsRole
            Me.xIt.Value = perm!xtsInitials
            
        End If
    
    dB.Close

End Sub
