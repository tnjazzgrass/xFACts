Private Sub impNB_Click()
Dim dB As Database
Dim nbx As Recordset
Dim rct As Long

    Set dB = CurrentDb
        Set nbx = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM nb_test_records;")
        rct = nbx!recs
    dB.Close
    
        If rct = 0 Then
        
            MsgBox "Please enter test account data before attempting to run the import.", vbCritical, "NO RECORDS FOUND"
            
        Else
            
            Select Case Me.runEnv
        
                Case 1
                
                    ReplaceBalances ("test")
            
                Case 2
                
                    ReplaceBalances ("stage")
            
                Case 3
                
                    ReplaceBalances ("prod")
                    
            End Select
            
        End If

End Sub

Private Sub nbBack_Click()

    DoCmd.Close acForm, "exp_nb_xml_main", acSaveNo
    DoCmd.OpenForm "fnncl_srvcs", acNormal

End Sub
