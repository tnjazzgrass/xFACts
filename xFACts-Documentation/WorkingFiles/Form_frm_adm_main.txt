Private Sub curQuery_Click()
Dim dB As Database
Dim strSQL As String, sql As String
Dim rs As ADODB.Recordset, rst As Recordset

    Set dB = CurrentDb
    
    dB.Execute ("DELETE * FROM dm_stat_db;")
    
    'Get Currently Running Queries
    strSQL = "SELECT sp.hostname"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.session_id"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , USER_NAME(r.USER_ID) AS Usr"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , DB_NAME(r.database_id) AS Db"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.status"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , SUBSTRING(qt.text, (r.statement_start_offset/2)+1"
    strSQL = strSQL & Chr(13) & Chr(10) & "                                , ((CASE r.statement_end_offset"
    strSQL = strSQL & Chr(13) & Chr(10) & "                                      WHEN -1 THEN DATALENGTH(qt.text)"
    strSQL = strSQL & Chr(13) & Chr(10) & "                                      ELSE r.statement_end_offset"
    strSQL = strSQL & Chr(13) & Chr(10) & "                                   END - r.statement_start_offset)/2) + 1) AS Sttmnt"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , d.name AS DBName"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , o.name AS ObjectName"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.cpu_time"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , CAST(r.start_time AS SMALLDATETIME) AS Start_Time"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , CAST(r.total_elapsed_time/1000/60.00 AS DECIMAL(8,2)) AS ElapsedMin"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.reads"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.writes"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.logical_reads"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.scheduler_id"
    strSQL = strSQL & Chr(13) & Chr(10) & "        , r.blocking_session_id"
    strSQL = strSQL & Chr(13) & Chr(10) & "FROM sys.dm_exec_requests r"
    strSQL = strSQL & Chr(13) & Chr(10) & "CROSS APPLY sys.dm_exec_sql_text(sql_handle) AS qt"
    strSQL = strSQL & Chr(13) & Chr(10) & "JOIN sys.sysprocesses sp ON r.session_id = sp.spid"
    strSQL = strSQL & Chr(13) & Chr(10) & "LEFT JOIN sys.objects o ON qt.objectid = o.OBJECT_ID"
    strSQL = strSQL & Chr(13) & Chr(10) & "LEFT JOIN sys.databases d ON qt.dbid = d.database_id"
    strSQL = strSQL & Chr(13) & Chr(10) & "WHERE r.session_id > 50"
    strSQL = strSQL & Chr(13) & Chr(10) & "ORDER BY ElapsedMin DESC;"

        OpenDMrecs rs, strSQL
    
        Do Until rs.EOF
            
            sql = "INSERT INTO dm_stat_db ([HOST], [SESSION], [USER], [DATABASE], [STATUS], [QUERY], [DB NAME], [OBJECT NAME], [CPU TIME], [START], [MINUTES], [READS], [WRITES], [LOGIC READS], [SCHEDULE ID], [BLOCKER ID]) "
            sql = sql & Chr(13) & Chr(10) & "VALUES (""" & Trim(rs!hostname) & """, " & rs!session_id & ", """ & rs!usr & """, """ & rs!dB & """, """ & rs!Status & """, """ & Trim(rs!Sttmnt) & """, """ & IIf(IsNull(rs!DBName), "NULL", rs!DBName) & """, """ & IIf(IsNull(rs!ObjectName), "NULL", rs!ObjectName) & """, " & rs!cpu_time & ", #" & rs!Start_Time & "#, " & Round(rs!ElapsedMin, 0) & ", " & rs!reads & ", " & rs!writes & ", " & rs!logical_reads & ", " & rs!scheduler_id & ", " & rs!blocking_session_id & ");"

            dB.Execute (sql)
            sql = ""
            
        rs.MoveNext
        Loop
        
        strSQL = ""
        rs.Close
        dB.Close
        
    DoCmd.OpenForm "frm_adm_db", acNormal
    DoCmd.Close acForm, "frm_adm_main", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim strSQL As String, sql As String
Dim rs As ADODB.Recordset, rst As Recordset

    Set dB = CurrentDb
    
        dB.Execute ("DELETE * FROM dm_stat_jobs;")
        dB.Execute ("DELETE * FROM dm_stat_strtgy;")
        dB.Execute ("DELETE * FROM dm_stat_btch;")
        dB.Execute ("DELETE * FROM dm_stat_bdl;")
        dB.Execute ("DELETE * FROM dm_stat_wrkgrp;")
    
    'Get Job Status
    strSQL = "SELECT * FROM vw_stat_jobs;"

        OpenDMrecs rs, strSQL
    
        Do Until rs.EOF
            
            sql = "INSERT INTO dm_stat_jobs (JOB_STATUS, JOBS) VALUES ('" & rs!JOB_STATUS & "', " & rs!JOBS & ");"
            dB.Execute (sql)
            sql = ""
            
        rs.MoveNext
        Loop
        rs.Close
        
            strSQL = "SELECT * FROM vw_stat_job_end;"

            OpenDMrecs rs, strSQL
            
                sql = "UPDATE dm_stat_jobs SET JOB_END = #" & rs!last_entry & "#;"
                dB.Execute (sql)
                sql = ""
        
        strSQL = ""
        rs.Close
        
    'Find stuck/stalled batches
    strSQL = "SELECT * FROM vw_stat_batches;"

        OpenDMrecs rs, strSQL
    
        Do Until rs.EOF
            
            sql = "INSERT INTO dm_stat_btch (BatchID, BatchStatus, ReleaseDate, BatchFile, Accounts) VALUES ('" & rs!new_bsnss_btch_id & "', '" & rs!new_bsnss_btch_stts_val_txt & "', '" & rs!new_bsnss_btch_rlsd_dt & "', '" & rs!new_bsnss_btch_upload_file_txt & "', " & IIf(IsNull(rs!new_bsnss_btch_cnsmr_accnt_actl_ttl_nmbr), 0, rs!new_bsnss_btch_cnsmr_accnt_actl_ttl_nmbr) & ");"
            dB.Execute (sql)
            sql = ""
            
        rs.MoveNext
        Loop
        
        strSQL = ""
        rs.Close
        
    'Find stuck/stalled bdl files
    strSQL = "SELECT * FROM vw_stat_bdl;"

        OpenDMrecs rs, strSQL
    
        Do Until rs.EOF
            
            sql = "INSERT INTO dm_stat_bdl (FileRegID, FileStatus, FileType, FileDtlID, FileDate, FileNameTxt, Records) VALUES ('" & rs!file_registry_id & "', '" & rs!file_stts_val_txt & "', '" & rs!file_typ_val_txt & "', '" & rs!file_rgstry_dtl_id & "', '" & rs!file_crt_dttm & "', '" & rs!file_name_full_txt & "', " & rs!file_rgstry_dtl_rec_ttl_cnt & ");"
            dB.Execute (sql)
            sql = ""
            
        rs.MoveNext
        Loop
        
        strSQL = ""
        rs.Close
        
    'Get Strategy Results
    strSQL = "SELECT * FROM vw_stat_strtgy;"
    
        OpenDMrecs rs, strSQL
            
            sql = "INSERT INTO dm_stat_strtgy (RUN_DATE,END_TIME,SUCCESSFUL,FAILURES,TOTAL_EXECS ) VALUES ( '" & rs!RUN_DATE & "', '" & rs!END_TIME & "', " & rs!SUCCESSFUL & ", " & rs!FAILURES & ", " & rs!TOTAL_EXECS & " );"
            dB.Execute (sql)
            sql = ""
        
        strSQL = ""
        rs.Close
        
    'Get Default Workgroup stats
    strSQL = "SELECT * FROM vw_stat_wrkgrp;"
    
        OpenDMrecs rs, strSQL
            
            sql = "INSERT INTO dm_stat_wrkgrp (RUN_DATE, EO, BD ) VALUES ( '" & rs!RunDate & "', " & IIf(IsNull(rs!FAM), 0, rs!FAM) & ", " & IIf(IsNull(rs!FAC), 0, rs!FAC) & ");"
            dB.Execute (sql)
            sql = ""
        
        strSQL = ""
        rs.Close

        'Update form
        Set rst = dB.OpenRecordset("SELECT JOBS AS pnd FROM dm_stat_jobs WHERE JOB_STATUS = 'PENDING';")
            Me.jPend.Value = IIf(rst.EOF, 0, rst!pnd)
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT JOBS AS strt FROM dm_stat_jobs WHERE JOB_STATUS = 'STARTED';")
            Me.jStrt.Value = IIf(rst.EOF, 0, rst!strt)
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT SUM(JOBS) AS comp FROM dm_stat_jobs WHERE JOB_STATUS IN ('FAILED','SUCCESS');")
            Me.jComp.Value = IIf(rst.EOF, 0, rst!comp)
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT MAX(JOB_END) AS ldt FROM dm_stat_jobs WHERE JOB_STATUS IN ('FAILED','SUCCESS');")
            Me.lstJob.Value = Format(rst!ldt, "h:nn AM/PM")
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT * FROM dm_stat_strtgy;")
            Me.sEnd.Value = rst!END_TIME
            Me.sFail.Value = Format(rst!FAILURES, "#,##0")
            Me.sTot.Value = Format(rst!TOTAL_EXECS, "#,##0")
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT * FROM dm_stat_wrkgrp;")
            Me.dft_1.Value = rst!EO
            Me.dft_3.Value = rst!BD
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT COUNT(*) AS stuck FROM dm_stat_btch WHERE BatchStatus = 'RELEASED';")
            Me.stkNB.Value = rst!stuck
        rst.Close
        
        Set rst = dB.OpenRecordset("SELECT COUNT(*) AS stuck FROM dm_stat_bdl WHERE FileDate > Format(Now(),'YYYY-MM-DD');")
            Me.stkBDL.Value = rst!stuck
        rst.Close
    
    dB.Close
    
End Sub
