Option Compare Database
Private Sub addAcctPay_Click()
Dim dB As Database
Dim rec As Recordset, rs As ADODB.Recordset
Dim strSQL As String, sql As String, err As Integer

    Set dB = CurrentDb

'ERROR CHECK
    err = 0
    
'ERROR  Consumer number not filled in
        If IsNull(Me.nAccount) = True Then
            MsgBox "Please enter an account number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
            err = err + 1
        End If
        
'ERROR  Payment amount not filled in
        If err = 0 And IsNull(Me.nAmount) = True Then
            MsgBox "Please enter a valid payment amount to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
            err = err + 1
        End If
        
'FORM IS FILLED OUT - TRY TO ADD PAYMENTS
        If err = 0 Then
        '   Look for eligible accounts under this consumer
            strSQL = "SELECT * FROM vw_xfacts_payable_accnts WHERE ACCOUNT_AGENCY_ID = " & Me.nAccount & ";"
            
                Select Case [Forms]![pay_batch_01_main]![runEnv]
                
                    Case 1
                        OpenDMrect rs, strSQL
                        
                    Case 2
                        OpenDMrecg rs, strSQL
                        
                    Case 3
                        OpenDMrecs rs, strSQL
                    
                End Select
 
 'ERROR         No eligible payments found
                If rs.EOF Then
                
                    err = err + 1
                    MsgBox "No account eligible for payment found for this account number in the environment selected. Please check the account number and try again.", vbOKOnly, "ACCOUNT NOT ELIGIBLE"
                    
                End If
 
 'ERROR         Creditor Overpayment
                If err = 0 And Me.nLocation = "CREDITOR" Then
                
                    If Me.nAmount > rs!account_curr_bal Then
                        
                        err = err + 1
                        MsgBox "The payment amount you entered would result in an overpayment and that is prohibited on creditor payments.", vbOKOnly, "CREDITOR OVERPAYMENT"
                    
                    End If
                
                End If
            
'NO ERRORS  Pull accounts into temp table to parse payments
            If err = 0 Then
        
                sql = "INSERT INTO payment_info (PayConsumerNumber, PayAccountNumber, PayAmt, PayType, PayLocation, PayEnterDate, PayTenderDate, PayMemoCode, PayComment, xUser, xType)"
                sql = sql & " SELECT '" & rs!CONSUMER_AGENCY_ID & "', " & rs!ACCOUNT_AGENCY_ID & ", " & Me.nAmount & ", '" & Me.nType & "', '" & Me.nLocation & "', '" & Format(Me.nDate, "yyyy-MM-dd") & "', '" & Format(Me.nDate, "yyyy-MM-dd") & "', '" & Me.nMemo & "', '" & [Forms]![pay_batch_01_main]![JIRA] & ": Payment batch', '" & [Forms]![pay_batch_01_main]![usr] & "', 'PAY';"
                dB.Execute sql
    
            End If
        
        End If
        
    rs.Close
    dB.Close
    
    Forms![pay_batch_01_main]![pay_batch_02_detail].Form.Requery
    DoCmd.Close acForm, "pay_batch_03_add_accnt", acSaveNo

End Sub

Private Sub canAcctPay_Click()

    DoCmd.Close acForm, "pay_batch_03_add_accnt", acSaveNo

End Sub
