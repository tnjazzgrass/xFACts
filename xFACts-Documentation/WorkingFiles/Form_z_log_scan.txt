Private Sub Form_Timer()
Dim dB As Database
Dim lck As Recordset

    Set dB = CurrentDb
    
    Set lck = dB.OpenRecordset("SELECT * FROM toolMaint")
    
        If lck!maintShutdown = -1 Then
            DoCmd.OpenForm "z_log_request", acNormal
        End If
    
    dB.Close

End Sub
