Private Sub admBack_Click()

    DoCmd.OpenForm "frm_adm_main", acNormal
    DoCmd.Close acForm, "frm_adm_stats", acSaveNo
    
End Sub
Private Sub Form_Load()
Dim dB As Database
Dim strSQL As String, sql As String
Dim rs As ADODB.Recordset, rst As Recordset

    Set dB = CurrentDb
    
    dB.Execute ("DELETE * FROM dm_stat_lkbk;")
    dB.Execute ("DELETE * FROM dm_stat_vdys;")

    'Get Validation Snapshot
    strSQL = "SELECT * FROM vw_stat_vdys_wk;"

        OpenDMrecs rs, strSQL

        Do Until rs.EOF

            sql = "INSERT INTO dm_stat_vdys (LOADED, ACCOUNTS, VALIDATED, AvgDayToVal, RptWk, LastPull) VALUES ('" & rs!LOADED & "', " & rs!Accounts & ", " & rs!VALIDATED & ", " & rs!ADTV & ", " & rs!RPT_WK & ", '" & Now() & "');"
            dB.Execute (sql)
            sql = ""

        rs.MoveNext
        Loop

        strSQL = ""
        rs.Close
    

    'Get last lookback job results
    strSQL = "SELECT * FROM vw_stat_lkbk;"

        OpenDMrecs rs, strSQL

        Do Until rs.EOF

            sql = "INSERT INTO dm_stat_lkbk (lst_rn, tags, schd, prcg, rlsd, cncd, sent) VALUES ('" & rs!RUN_DATE & "', " & rs!tags & ", " & rs!schd & ", " & rs!prcg & ", " & rs!rlsd & ", " & rs!cncd & ", " & rs!sent & ");"
            dB.Execute (sql)
            sql = ""

        rs.MoveNext
        Loop

        strSQL = ""
        rs.Close
        
    Me.frm_adm_stats_wkly.Requery
    Me.Requery
    
    dB.Close
    
End Sub
