Private Sub BCBSimpNew_Click()
Dim fs As Object
Dim dB As Database
Dim err As Integer, cnum As String
Dim fName As String, strSQL As String, sql As String, strSQL2 As String
Dim fdate As Recordset, famt As Recordset, rs As ADODB.Recordset, rsc As Recordset, rss As ADODB.Recordset

    err = 0
    Set dB = CurrentDb
    
    dB.Execute ("DELETE * FROM bcbs_accnts;")
    
    '   Verify records entered into table
    '   Check values in the amounts fields
        Set famt = dB.OpenRecordset("SELECT * FROM payBCBS_stg WHERE [Payment Amount] < 0 OR [Adjustment Amount] < 0;")
        If famt.EOF Then
        
            Else
            
                MsgBox "The import file has records that have an invalid Payment or Adjustment amount (negative amounts). This process can only be used to apply payments and write-off adjustments, please verify the amounts on the spreadsheet and re-process a corrected file.", vbExclamation, "ERROR: INVALID AMOUNT ENTERED"
                err = err + 1
                
        End If
        
    '   Check values in the filename field
        If IsNull(Me.impFile) Then
        
            MsgBox "Please provide a filename for this import.", vbExclamation, "FILENAME IS BLANK"
            err = err + 1
            
        Else
        
            dB.Execute ("UPDATE payBCBS_stg SET BCBS_FILE_NM = '" & Me.impFile & "';")
                
        End If
        
    '   Check for a file date
        Set fdate = dB.OpenRecordset("SELECT * FROM payBCBS_stg WHERE [File Date] IS NULL;")
        If fdate.EOF Then
        
            Else
                MsgBox "The import file has records that are missing a File Date. Today's date will be used as the Entered/Tendered date for these records.", vbInformation, "WARNING: DEFAULT DATE USED"
                dB.Execute ("UPDATE payBCBS_stg SET [File Date] = Format(Now(),""yyyy-mm-dd"") WHERE [File Date] IS NULL;")
        End If
        
'   If all records are valid and ready for processing, insert into pay table
    If err = 0 Then
    
        ExecDMcmd ("DELETE FROM xFACts_BCBS_cred_ref;")
        
        Set rsc = dB.OpenRecordset("SELECT * FROM payBCBS_stg;")
        Do Until rsc.EOF
            
            strSQL = "INSERT INTO xFACts_BCBS_cred_ref (claim_num) VALUES ('" & rsc![Claim Number] & "');"
            ExecDMcmd (strSQL)
            
        rsc.MoveNext
        Loop
        
            strSQL = "SELECT * FROM Applications.dbo.vw_bcbs_accnts;"
            
            'Debug.Print strSQL
            
            OpenDMrecs rs, strSQL
            
            If rs.EOF Then
            
                dB.Execute ("UPDATE payBCBS_stg SET DM_RES = 'FAILED: Account not found', PROC_DT = '" & Now() & "';")
                
            Else
                
                Do Until rs.EOF
                
                    sql = "INSERT INTO bcbs_accnts (cnsmr_accnt_crdtr_rfrnc_id_txt, cnsmr_accnt_idntfr_agncy_id, cnsmr_idntfr_agncy_id, crdtr_shrt_nm, cnsmr_accnt_id, cnsmr_id, cnsmr_accnt_plcmnt_date, cnsmr_accnt_bal_amnt, acs_tag)"
                    sql = sql & " VALUES ('" & rs!cnsmr_accnt_crdtr_rfrnc_id_txt & "','" & rs!cnsmr_accnt_idntfr_agncy_id & "','" & rs!cnsmr_idntfr_agncy_id & "','" & rs!crdtr_shrt_nm & "','" & rs!cnsmr_accnt_id & "','" & rs!cnsmr_id & "','" & rs!cnsmr_accnt_plcmnt_date & "'," & rs!cnsmr_accnt_bal_amnt & ",'" & rs!acs_tag & "');"
                    dB.Execute (sql)
                    
                    sql = ""
                
                rs.MoveNext
                Loop
            
            End If
        
        sql = ""
        cnum = ""
        strSQL = ""
        strSQL2 = ""
        
        DoCmd.OpenQuery "bcbs_upd_qry_01a_add_unfound", acViewNormal, acEdit
        dB.Execute ("INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('impBCBSpays', 'N/A', '" & Environ("username") & "', Now())")
        MsgBox "The specified records file have been verified and are ready to be posted in DM.", vbOKOnly, "RECORD LOAD COMPLETE"
    
    Else
        
        dB.Execute ("DELETE * FROM payBCBS_stg;")
            
    End If


    
 '  Verify records in stage screen
    DoCmd.OpenQuery "bcbs_upd_qry_01a_add_account", acViewNormal, acEdit
    DoCmd.OpenQuery "bcbs_upd_qry_01b_add_miss_accnt", acViewNormal, acEdit
    
        dB.Execute ("DELETE * FROM bcbs_dupes;")
        DoCmd.OpenQuery "bcbs_add_qry02_dupes", acViewNormal, acEdit
        DoCmd.OpenQuery "bcbs_upd_qry_01_mark_dupes", acViewNormal, acEdit
        DoCmd.OpenQuery "bcbs_upd_qry_02_mark_clsd", acViewNormal, acEdit
        DoCmd.OpenQuery "bcbs_upd_qry_03_mark_ovrpy", acViewNormal, acEdit
        DoCmd.OpenQuery "bcbs_upd_qry_04_mark_notfnd", acViewNormal, acEdit
        DoCmd.OpenQuery "bcbs_upd_qry_05_mark_ready", acViewNormal, acEdit

    DoCmd.OpenForm "bcbs_01main", acNormal
    DoCmd.Close acForm, "bcbs_02import", acSaveNo
    
    dB.Close

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close

End Sub
