Option Compare Database

Private Sub backPays_Click()
    
    DoCmd.OpenForm "frm_adm_main", acNormal
    DoCmd.Close acForm, "frm_adm_lock", acSaveNo

End Sub

Private Sub isLocked_AfterUpdate()
Dim dB As Database

    Set dB = CurrentDb
    
    dB.Execute ("UPDATE toolMaint SET maintShutdown = 0;")
    
    dB.Close

End Sub

Private Sub shutdown_Click()
Dim dB As Database

    Set dB = CurrentDb
    
    dB.Execute ("UPDATE toolMaint SET maintShutdown = -1, maintLocked = -1;")
    
    dB.Close
    
    MsgBox "The Apps Toolkit is in Forced Shutdown mode.", vbOKOnly, "FORCED SHUTDOWN REQUeSTED"

End Sub
