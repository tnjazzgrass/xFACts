Attribute VB_Name = "Form_frm_pay_add_accnt"

Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Compare Database

Private Sub addActPay_Click()
Dim sql As String
Dim err As Integer
Dim rec As Recordset
Dim dB As Database
Dim look As Recordset
Dim curr_bal As Double

'   Error Check - make sure form is filled out
    err = 0
    
    If IsNull(Me.nAccount) = True Then
        MsgBox "Please enter an account number to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.nAmount) = True Then
        MsgBox "Please enter a valid payment amount to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If
    If IsNull(Me.nComment) = True Then
        MsgBox "Please enter a ticket number or payment comment to proceed.", vbOKOnly, "MISSING OR INVALID PAYMENT DATA"
        err = err + 1
    End If

    If err = 0 Then
    
        Set dB = CurrentDb
        
        Set look = dB.OpenRecordset("SELECT cnsmr_idntfr_agncy_id FROM dbo_vw_Apps Toolkit_cnsmr_accnts WHERE cnsmr_accnt_idntfr_agncy_id = '" & Me.nAccount & "';")
        
        If look.EOF Then
        
            MsgBox "This app cannot be used to process payments for this account. Please use Debt Manager to post payments on consumers with fewer than 100 accounts.", vbOKOnly, "NO RECORD FOUND"
            
        Else
            
            Set rec = dB.OpenRecordset("SELECT ACCOUNT_AGENCY_ID, account_curr_bal, account_status_tag FROM dbo_vw_Apps Toolkit_accnt_smmry WHERE ACCOUNT_AGENCY_ID = '" & Me.nAccount & "';")
            
                Select Case rec!account_status_tag
                    Case "TA_CSCLS", "TA_CSPIF", "TA_CSRET", "TA_CSRTR", "TA_CSSIF"
                        err = err + 1
                        MsgBox "This account has a status tag which makes it ineligible for payments. Please choose another account.", vbOKOnly, "ACCOUNT CLOSED"
                End Select
                
                    If err = 0 Then
                        If rec!account_curr_bal < Me.nAmount Then
                                err = err + 1
                                MsgBox "The amount you have entered for this payment exceeds the current balance for this account. Please enter a lower payment amount or choose another account.", vbOKOnly, "OVERPAYMENT"
                        End If
                    End If
            
                If err = 0 Then
                
                    sql = "INSERT INTO payment_info (PayConsumerNumber, PayAccountNumber, PayAmt, PayType, PayLocation, PayEnterDate, PayTenderDate, PayMemoCode, PayComment, xUser, xType)"
                    sql = sql & " SELECT " & look!cnsmr_idntfr_agncy_id & ", " & Me.nAccount & ", " & Me.nAmount & ", '" & Me.nType & "', '" & Me.nLocation & "', '" & Format(Me.nDate, "yyyy-MM-dd") & "', '" & Format(Me.nDate, "yyyy-MM-dd") & "', '" & Me.nMemo & "', '" & Me.nComment & "', '" & [Forms]![Apps Toolkit]![xIt] & "', 'APY';"
                    
                    dB.Execute sql
                    
                End If
            
        End If
        
        DoCmd.OpenForm "frm_pay_rec_" & [Forms]![Apps Toolkit]![xIt], acNormal
        DoCmd.Close acForm, "frm_pay_add_accnt", acSaveNo
    
    End If
End Sub

Private Sub canActPay_Click()

    DoCmd.Close acForm, Me.Form.Name, acSaveNo
    DoCmd.OpenForm "frm_pay_rec_" & [Forms]![Apps Toolkit]![xIt], acNormal

End Sub
