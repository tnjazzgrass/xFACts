Attribute VB_PredeclaredId = True

Option Compare Database
Private Sub backAddrUpdt_Click()

    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_addr_updts", acSaveNo

End Sub

Private Sub Form_Load()
Dim dB As Database
Dim u As Recordset

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT xtsRole, xtsInitials FROM xtsUsers WHERE xtsUsername = '" & Environ("username") & "';")
    
        Me.usr.Value = u!xtsInitials
    
    dB.Close
    
    Me.bdl_addrss_updt_dtl.Requery
    
End Sub
Private Sub pstAddrUpdt_Click()
Dim dB As Database
Dim tgs As Recordset

    Set dB = CurrentDb
    Set tgs = dB.OpenRecordset("SELECT COUNT(*) AS recs FROM bdl_addrss_updt WHERE bdl_usr = '" & Me.usr.Value & "' AND bdl_proc = 'N';")
    
        If tgs!recs = 0 Then
        
            MsgBox "Please enter new updates before attempting to run the BDL.", vbCritical, "NO UNPROCESSED RECORDS FOUND"
            
        Else

            If Me.JIRA = "SD-" Or Left(Me.JIRA, 3) <> "SD-" Then
                
                    MsgBox "Please enter a JIRA IT Ticket number to proceed.", vbCritical, "INVALID JIRA NUMBER"
                    
            Else
            
                Select Case Me.runEnv
            
                    Case 1
                    
                        PostAddrUpdts ("test")
                
                    Case 2
                    
                        PostAddrUpdts ("stage")
                
                    Case 3
                    
                        PostAddrUpdts ("prod")
                        
                End Select
            
            End If
            
        End If
        
    dB.Close
    DoCmd.OpenForm "BDL Files", acNormal
    DoCmd.Close acForm, "bdl_pst_addr_updts", acSaveNo

End Sub

Private Sub runMR_AfterUpdate()

    If Me.runMR = -1 Then
    
        MsgBox "Please Note: By checking this box, you will be running the Remove Mail Return API on all consumer addresses in your file. This may take some time to run, depending on how many records you have. If you're updating over 1,000 addresses, please don't run this in PROD during working hours.", vbInformation, "INCLUDE MAIL RETURN API"
    
    End If
    
End Sub
