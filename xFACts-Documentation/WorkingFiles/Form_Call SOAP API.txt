Private Sub SubSOAP_Click()
Dim sURL As String
Dim sEnv As String
Dim xmlhtp As New XMLHTTP60

    sURL = Me.APIurl
    sEnv = Me.APIbody
    
        xmlhtp.Open "POST", sURL, False
        xmlhtp.setRequestHeader "Content-Type", "text/xml"
        xmlhtp.setRequestHeader "Host", Me.ServURL
        xmlhtp.sEnd sEnv
    
    MsgBox xmlhtp.responseText, vbInformation, "WEBSERVICE RESPONSE"

End Sub
