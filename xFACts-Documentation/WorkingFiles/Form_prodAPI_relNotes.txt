Private Sub canRelProdNotes_Click()
    
    DoCmd.Close acForm, "prodAPI_relNotes", acSaveNo
    [Forms]![Call REST API].Visible = True

End Sub

Private Sub relProdNotes_Click()
Dim dB As Database
Dim httpReq As New XMLHTTP60
Dim response As String, jBody As String, sql As String
Dim sURL As String, sUser As String, sPass As String, sAuth As String

    Set dB = CurrentDb
    
        Set rst1 = dB.OpenRecordset("SELECT MAX(toolDate) AS lst FROM toolUse WHERE toolFunction = 'relNotes' AND toolEnv = 'PROD';")
        
        If DateDiff("n", rst1!lst, Now()) < 5 Then
        
            MsgBox "This API was called less than 5 minutes ago, please wait before using this function.", vbInformation, "API RECENTLY CALLED"
            
        Else
            
            sURL = "http://dm-prod-app:8080/dm-rest-services/api/scheduledjobs/RELEASE_DOC_REQUESTS"
             
                httpReq.Open "POST", sURL, False, [Forms]![Call REST API]![APIuser], [Forms]![Call REST API]![APIpass]
                httpReq.setRequestHeader "Content-Type", "application/vnd.fico.dm.v1+json"
                httpReq.setRequestHeader "Authorization", [Forms]![Call REST API]![authCode]
                httpReq.sEnd
                
                response = httpReq.responseText
               
            dB.Execute "INSERT INTO toolUse (toolFunction, toolEnv, toolUser, toolDate) VALUES ('relNotes', 'PROD', '" & Environ("username") & "', Now())"
            
            MsgBox response, vbInformation, "PRODUCTION WEBSERVICE RESPONSE"
        
            DoCmd.Close acForm, "prodAPI_relNotes", acSaveNo
            [Forms]![Call REST API].Visible = True
            
        End If
    

                                   
End Sub
