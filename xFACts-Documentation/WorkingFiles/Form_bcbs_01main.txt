Private Sub opnImpBCBS_Click()
Dim dB As Database

    Set dB = CurrentDb
        dB.Execute ("DELETE * FROM payBCBS_stg;")
    dB.Close

    DoCmd.OpenForm "bcbs_02import", acNormal
    DoCmd.Close acForm, "bcbs_01main", acSaveNo

End Sub

Private Sub opnPostBCBS_Click()
Dim dB As Database
Dim rec As Recordset, dtChk As Recordset

    DoCmd.OpenForm "Apps Toolkit", acNormal
    Forms![Apps Toolkit].Visible = False
    
    Set dB = CurrentDb
    Set rec = dB.OpenRecordset("SELECT * FROM payBCBS_stg WHERE PROC='N';")
    
        If rec.EOF Then
        
            MsgBox "Please import new records to post before continuing with the process.", vbOKOnly, "NO NEW RECORDS TO POST"
            
        Else
                
            DoCmd.OpenForm "bcbs_04postPays", acNormal
            DoCmd.Close acForm, "bcbs_01main", acSaveNo
                
        End If
        
    dB.Close

End Sub
