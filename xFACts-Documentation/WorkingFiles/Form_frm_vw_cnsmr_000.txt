Attribute VB_Name = "Form_frm_vw_cnsmr_000"

Attribute VB_PredeclaredId = True

Option Compare Database
Private Sub canSearch_Click()
Dim dB As Database

    Set dB = CurrentDb
    Set u = dB.OpenRecordset("SELECT * FROM toolUsers WHERE toolUsername = '" & Environ("username") & "';")
    dB.Execute ("DELETE * FROM xfacts_cnsmr_smmry WHERE xUser = '" & u!toolInitials & "';")
    dB.Execute ("DELETE * FROM xfacts_accnt_smmry WHERE xUser = '" & u!toolInitials & "';")
    dB.Execute ("DELETE * FROM xfacts_ar_events WHERE xUser = '" & u!toolInitials & "';")
    dB.Close

    DoCmd.OpenForm "frm_lkp_cnsmr_000"
    DoCmd.Close acForm, Me.Form.Name, acSaveNo
    
End Sub
